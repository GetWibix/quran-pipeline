بِسۡمِ ٱللَّهِ ٱلرَّحۡمَـٰنِ ٱلرَّحِیمِ وَٱلصَّـٰۤفَّـٰتِ صَفࣰّا
2
فَٱلزَّ ٰ⁠جِرَ ٰ⁠تِ زَجۡرࣰا
3
فَٱلتَّـٰلِیَـٰتِ ذِكۡرًا
4
إِنَّ إِلَـٰهَكُمۡ لَوَ ٰ⁠حِدࣱ
5
رَّبُّ ٱلسَّمَـٰوَ ٰ⁠تِ وَٱلۡأَرۡضِ وَمَا بَیۡنَهُمَا وَرَبُّ ٱلۡمَشَـٰرِقِ
6
إِنَّا زَیَّنَّا ٱلسَّمَاۤءَ ٱلدُّنۡیَا بِزِینَةٍ ٱلۡكَوَاكِبِ
7
وَحِفۡظࣰا مِّن كُلِّ شَیۡطَـٰنࣲ مَّارِدࣲ
8
لَّا یَسَّمَّعُونَ إِلَى ٱلۡمَلَإِ ٱلۡأَعۡلَىٰ وَیُقۡذَفُونَ مِن كُلِّ جَانِبࣲ

دُحُورࣰاۖ وَلَهُمۡ عَذَابࣱ وَاصِبٌ

إِلَّا مَنۡ خَطِفَ ٱلۡخَطۡفَةَ فَأَتۡبَعَهُۥ شِهَابࣱ ثَاقِبࣱ
11
فَٱسۡتَفۡتِهِمۡ أَهُمۡ أَشَدُّ خَلۡقًا أَم مَّنۡ خَلَقۡنَاۤۚ إِنَّا خَلَقۡنَـٰهُم مِّن طِینࣲ لَّازِبِۭ

بَلۡ عَجِبۡتَ وَیَسۡخَرُونَ
13
وَإِذَا ذُكِّرُوا۟ لَا یَذۡكُرُونَ

وَإِذَا رَأَوۡا۟ ءَایَةࣰ یَسۡتَسۡخِرُونَ

وَقَالُوۤا۟ إِنۡ هَـٰذَاۤ إِلَّا سِحۡرࣱ مُّبِینٌ
16
أَءِذَا مِتۡنَا وَكُنَّا تُرَابࣰا وَعِظَـٰمًا أَءِنَّا لَمَبۡعُوثُونَ

أَوَءَابَاۤؤُنَا ٱلۡأَوَّلُونَ

قُلۡ نَعَمۡ وَأَنتُمۡ دَ ٰ⁠خِرُونَ
19
فَإِنَّمَا هِیَ زَجۡرَةࣱ وَ ٰ⁠حِدَةࣱ فَإِذَا هُمۡ یَنظُرُونَ

وَقَالُوا۟ یَـٰوَیۡلَنَا هَـٰذَا یَوۡمُ ٱلدِّینِ

هَـٰذَا یَوۡمُ ٱلۡفَصۡلِ ٱلَّذِی كُنتُم بِهِۦ تُكَذِّبُونَ
22
۞ ٱحۡشُرُوا۟ ٱلَّذِینَ ظَلَمُوا۟ وَأَزۡوَ ٰ⁠جَهُمۡ وَمَا كَانُوا۟ یَعۡبُدُونَ
23
مِن دُونِ ٱللَّهِ فَٱهۡدُوهُمۡ إِلَىٰ صِرَ ٰ⁠طِ ٱلۡجَحِیمِ

وَقِفُوهُمۡۖ إِنَّهُم مَّسۡـُٔولُونَ

مَا لَكُمۡ لَا تَنَاصَرُونَ

بَلۡ هُمُ ٱلۡیَوۡمَ مُسۡتَسۡلِمُونَ

وَأَقۡبَلَ بَعۡضُهُمۡ عَلَىٰ بَعۡضࣲ یَتَسَاۤءَلُونَ
28
قَالُوۤا۟ إِنَّكُمۡ كُنتُمۡ تَأۡتُونَنَا عَنِ ٱلۡیَمِینِ
29
قَالُوا۟ بَل لَّمۡ تَكُونُوا۟ مُؤۡمِنِینَ
30
وَمَا كَانَ لَنَا عَلَیۡكُم مِّن سُلۡطَـٰنِۭۖ بَلۡ كُنتُمۡ قَوۡمࣰا طَـٰغِینَ
31
فَحَقَّ عَلَیۡنَا قَوۡلُ رَبِّنَاۤۖ إِنَّا لَذَاۤىِٕقُونَ
32
فَأَغۡوَیۡنَـٰكُمۡ إِنَّا كُنَّا غَـٰوِینَ
33
فَإِنَّهُمۡ یَوۡمَىِٕذࣲ فِی ٱلۡعَذَابِ مُشۡتَرِكُونَ
34
إِنَّا كَذَ ٰ⁠لِكَ نَفۡعَلُ بِٱلۡمُجۡرِمِینَ
35
إِنَّهُمۡ كَانُوۤا۟ إِذَا قِیلَ لَهُمۡ لَاۤ إِلَـٰهَ إِلَّا ٱللَّهُ یَسۡتَكۡبِرُونَ
36
وَیَقُولُونَ أَىِٕنَّا لَتَارِكُوۤا۟ ءَالِهَتِنَا لِشَاعِرࣲ مَّجۡنُونِۭ

بَلۡ جَاۤءَ بِٱلۡحَقِّ وَصَدَّقَ ٱلۡمُرۡسَلِینَ
38
إِنَّكُمۡ لَذَاۤىِٕقُوا۟ ٱلۡعَذَابِ ٱلۡأَلِیمِ
39
وَمَا تُجۡزَوۡنَ إِلَّا مَا كُنتُمۡ تَعۡمَلُونَ
40
إِلَّا عِبَادَ ٱللَّهِ ٱلۡمُخۡلَصِینَ
41
أُو۟لَـٰۤىِٕكَ لَهُمۡ رِزۡقࣱ مَّعۡلُومࣱ
42
فَوَ ٰ⁠كِهُ وَهُم مُّكۡرَمُونَ
43
فِی جَنَّـٰتِ ٱلنَّعِیمِ
44
عَلَىٰ سُرُرࣲ مُّتَقَـٰبِلِینَ
45
یُطَافُ عَلَیۡهِم بِكَأۡسࣲ مِّن مَّعِینِۭ

بَیۡضَاۤءَ لَذَّةࣲ لِّلشَّـٰرِبِینَ
47
لَا فِیهَا غَوۡلࣱ وَلَا هُمۡ عَنۡهَا یُنزَفُونَ

وَعِندَهُمۡ قَـٰصِرَ ٰ⁠تُ ٱلطَّرۡفِ عِینࣱ

كَأَنَّهُنَّ بَیۡضࣱ مَّكۡنُونࣱ

فَأَقۡبَلَ بَعۡضُهُمۡ عَلَىٰ بَعۡضࣲ یَتَسَاۤءَلُونَ
51
قَالَ قَاۤىِٕلࣱ مِّنۡهُمۡ إِنِّی كَانَ لِی قَرِینࣱ
52
یَقُولُ أَءِنَّكَ لَمِنَ ٱلۡمُصَدِّقِینَ
53
أَءِذَا مِتۡنَا وَكُنَّا تُرَابࣰا وَعِظَـٰمًا أَءِنَّا لَمَدِینُونَ

قَالَ هَلۡ أَنتُم مُّطَّلِعُونَ

فَٱطَّلَعَ فَرَءَاهُ فِی سَوَاۤءِ ٱلۡجَحِیمِ

قَالَ تَٱللَّهِ إِن كِدتَّ لَتُرۡدِینِ
57
وَلَوۡلَا نِعۡمَةُ رَبِّی لَكُنتُ مِنَ ٱلۡمُحۡضَرِینَ

أَفَمَا نَحۡنُ بِمَیِّتِینَ
59
إِلَّا مَوۡتَتَنَا ٱلۡأُولَىٰ وَمَا نَحۡنُ بِمُعَذَّبِینَ

إِنَّ هَـٰذَا لَهُوَ ٱلۡفَوۡزُ ٱلۡعَظِیمُ

لِمِثۡلِ هَـٰذَا فَلۡیَعۡمَلِ ٱلۡعَـٰمِلُونَ
62
أَذَ ٰ⁠لِكَ خَیۡرࣱ نُّزُلًا أَمۡ شَجَرَةُ ٱلزَّقُّومِ

إِنَّا جَعَلۡنَـٰهَا فِتۡنَةࣰ لِّلظَّـٰلِمِینَ
64
إِنَّهَا شَجَرَةࣱ تَخۡرُجُ فِیۤ أَصۡلِ ٱلۡجَحِیمِ

طَلۡعُهَا كَأَنَّهُۥ رُءُوسُ ٱلشَّیَـٰطِینِ

فَإِنَّهُمۡ لَـَٔاكِلُونَ مِنۡهَا فَمَالِـُٔونَ مِنۡهَا ٱلۡبُطُونَ
67
ثُمَّ إِنَّ لَهُمۡ عَلَیۡهَا لَشَوۡبࣰا مِّنۡ حَمِیمࣲ

ثُمَّ إِنَّ مَرۡجِعَهُمۡ لَإِلَى ٱلۡجَحِیمِ

إِنَّهُمۡ أَلۡفَوۡا۟ ءَابَاۤءَهُمۡ ضَاۤلِّینَ

فَهُمۡ عَلَىٰۤ ءَاثَـٰرِهِمۡ یُهۡرَعُونَ

وَلَقَدۡ ضَلَّ قَبۡلَهُمۡ أَكۡثَرُ ٱلۡأَوَّلِینَ

وَلَقَدۡ أَرۡسَلۡنَا فِیهِم مُّنذِرِینَ

فَٱنظُرۡ كَیۡفَ كَانَ عَـٰقِبَةُ ٱلۡمُنذَرِینَ

إِلَّا عِبَادَ ٱللَّهِ ٱلۡمُخۡلَصِینَ
75
وَلَقَدۡ نَادَىٰنَا نُوحࣱ فَلَنِعۡمَ ٱلۡمُجِیبُونَ
76
وَنَجَّیۡنَـٰهُ وَأَهۡلَهُۥ مِنَ ٱلۡكَرۡبِ ٱلۡعَظِیمِ

وَجَعَلۡنَا ذُرِّیَّتَهُۥ هُمُ ٱلۡبَاقِینَ

وَتَرَكۡنَا عَلَیۡهِ فِی ٱلۡـَٔاخِرِینَ
79
سَلَـٰمٌ عَلَىٰ نُوحࣲ فِی ٱلۡعَـٰلَمِینَ
80
إِنَّا كَذَ ٰ⁠لِكَ نَجۡزِی ٱلۡمُحۡسِنِینَ

إِنَّهُۥ مِنۡ عِبَادِنَا ٱلۡمُؤۡمِنِینَ

ثُمَّ أَغۡرَقۡنَا ٱلۡـَٔاخَرِینَ
83
۞ وَإِنَّ مِن شِیعَتِهِۦ لَإِبۡرَ ٰ⁠هِیمَ

إِذۡ جَاۤءَ رَبَّهُۥ بِقَلۡبࣲ سَلِیمٍ

إِذۡ قَالَ لِأَبِیهِ وَقَوۡمِهِۦ مَاذَا تَعۡبُدُونَ

أَىِٕفۡكًا ءَالِهَةࣰ دُونَ ٱللَّهِ تُرِیدُونَ

فَمَا ظَنُّكُم بِرَبِّ ٱلۡعَـٰلَمِینَ

فَنَظَرَ نَظۡرَةࣰ فِی ٱلنُّجُومِ
89
فَقَالَ إِنِّی سَقِیمࣱ
90
فَتَوَلَّوۡا۟ عَنۡهُ مُدۡبِرِینَ

فَرَاغَ إِلَىٰۤ ءَالِهَتِهِمۡ فَقَالَ أَلَا تَأۡكُلُونَ

مَا لَكُمۡ لَا تَنطِقُونَ

فَرَاغَ عَلَیۡهِمۡ ضَرۡبَۢا بِٱلۡیَمِینِ
94
فَأَقۡبَلُوۤا۟ إِلَیۡهِ یَزِفُّونَ

قَالَ أَتَعۡبُدُونَ مَا تَنۡحِتُونَ

وَٱللَّهُ خَلَقَكُمۡ وَمَا تَعۡمَلُونَ
97
قَالُوا۟ ٱبۡنُوا۟ لَهُۥ بُنۡیَـٰنࣰا فَأَلۡقُوهُ فِی ٱلۡجَحِیمِ
98
فَأَرَادُوا۟ بِهِۦ كَیۡدࣰا فَجَعَلۡنَـٰهُمُ ٱلۡأَسۡفَلِینَ
99
وَقَالَ إِنِّی ذَاهِبٌ إِلَىٰ رَبِّی سَیَهۡدِینِ
100
رَبِّ هَبۡ لِی مِنَ ٱلصَّـٰلِحِینَ
101
فَبَشَّرۡنَـٰهُ بِغُلَـٰمٍ حَلِیمࣲ

فَلَمَّا بَلَغَ مَعَهُ ٱلسَّعۡیَ قَالَ یَـٰبُنَیَّ إِنِّیۤ أَرَىٰ فِی ٱلۡمَنَامِ أَنِّیۤ أَذۡبَحُكَ فَٱنظُرۡ مَاذَا تَرَىٰۚ قَالَ یَـٰۤأَبَتِ ٱفۡعَلۡ مَا تُؤۡمَرُۖ سَتَجِدُنِیۤ إِن شَاۤءَ ٱللَّهُ مِنَ ٱلصَّـٰبِرِینَ
103
فَلَمَّاۤ أَسۡلَمَا وَتَلَّهُۥ لِلۡجَبِینِ
104
وَنَـٰدَیۡنَـٰهُ أَن یَـٰۤإِبۡرَ ٰ⁠هِیمُ
105
قَدۡ صَدَّقۡتَ ٱلرُّءۡیَاۤۚ إِنَّا كَذَ ٰ⁠لِكَ نَجۡزِی ٱلۡمُحۡسِنِینَ
106
إِنَّ هَـٰذَا لَهُوَ ٱلۡبَلَـٰۤؤُا۟ ٱلۡمُبِینُ
107
وَفَدَیۡنَـٰهُ بِذِبۡحٍ عَظِیمࣲ
108
وَتَرَكۡنَا عَلَیۡهِ فِی ٱلۡـَٔاخِرِینَ
109
سَلَـٰمٌ عَلَىٰۤ إِبۡرَ ٰ⁠هِیمَ
110
كَذَ ٰ⁠لِكَ نَجۡزِی ٱلۡمُحۡسِنِینَ

إِنَّهُۥ مِنۡ عِبَادِنَا ٱلۡمُؤۡمِنِینَ
112
وَبَشَّرۡنَـٰهُ بِإِسۡحَـٰقَ نَبِیࣰّا مِّنَ ٱلصَّـٰلِحِینَ
113
وَبَـٰرَكۡنَا عَلَیۡهِ وَعَلَىٰۤ إِسۡحَـٰقَۚ وَمِن ذُرِّیَّتِهِمَا مُحۡسِنࣱ وَظَالِمࣱ لِّنَفۡسِهِۦ مُبِینࣱ

وَلَقَدۡ مَنَنَّا عَلَىٰ مُوسَىٰ وَهَـٰرُونَ
115
وَنَجَّیۡنَـٰهُمَا وَقَوۡمَهُمَا مِنَ ٱلۡكَرۡبِ ٱلۡعَظِیمِ
116
وَنَصَرۡنَـٰهُمۡ فَكَانُوا۟ هُمُ ٱلۡغَـٰلِبِینَ

وَءَاتَیۡنَـٰهُمَا ٱلۡكِتَـٰبَ ٱلۡمُسۡتَبِینَ

وَهَدَیۡنَـٰهُمَا ٱلصِّرَ ٰ⁠طَ ٱلۡمُسۡتَقِیمَ

وَتَرَكۡنَا عَلَیۡهِمَا فِی ٱلۡـَٔاخِرِینَ
120
سَلَـٰمٌ عَلَىٰ مُوسَىٰ وَهَـٰرُونَ
121
إِنَّا كَذَ ٰ⁠لِكَ نَجۡزِی ٱلۡمُحۡسِنِینَ

إِنَّهُمَا مِنۡ عِبَادِنَا ٱلۡمُؤۡمِنِینَ

وَإِنَّ إِلۡیَاسَ لَمِنَ ٱلۡمُرۡسَلِینَ

إِذۡ قَالَ لِقَوۡمِهِۦۤ أَلَا تَتَّقُونَ

أَتَدۡعُونَ بَعۡلࣰا وَتَذَرُونَ أَحۡسَنَ ٱلۡخَـٰلِقِینَ
126
