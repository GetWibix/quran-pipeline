الآية
تنزيل
1
بِسۡمِ ٱللَّهِ ٱلرَّحۡمَـٰنِ ٱلرَّحِیمِ سَأَلَ سَاۤىِٕلُۢ بِعَذَابࣲ وَاقِعࣲ
2
لِّلۡكَـٰفِرِینَ لَیۡسَ لَهُۥ دَافِعࣱ
3
مِّنَ ٱللَّهِ ذِی ٱلۡمَعَارِجِ
4
تَعۡرُجُ ٱلۡمَلَـٰۤىِٕكَةُ وَٱلرُّوحُ إِلَیۡهِ فِی یَوۡمࣲ كَانَ مِقۡدَارُهُۥ خَمۡسِینَ أَلۡفَ سَنَةࣲ
5
فَٱصۡبِرۡ صَبۡرࣰا جَمِیلًا
6
إِنَّهُمۡ یَرَوۡنَهُۥ بَعِیدࣰا
7
وَنَرَىٰهُ قَرِیبࣰا
8
یَوۡمَ تَكُونُ ٱلسَّمَاۤءُ كَٱلۡمُهۡلِ
9
وَتَكُونُ ٱلۡجِبَالُ كَٱلۡعِهۡنِ
10
وَلَا یَسۡـَٔلُ حَمِیمٌ حَمِیمࣰا
11
یُبَصَّرُونَهُمۡۚ یَوَدُّ ٱلۡمُجۡرِمُ لَوۡ یَفۡتَدِی مِنۡ عَذَابِ یَوۡمِىِٕذِۭ بِبَنِیهِ
12
وَصَـٰحِبَتِهِۦ وَأَخِیهِ
13
وَفَصِیلَتِهِ ٱلَّتِی تُـٔۡوِیهِ
14
وَمَن فِی ٱلۡأَرۡضِ جَمِیعࣰا ثُمَّ یُنجِیهِ
15
كَلَّاۤۖ إِنَّهَا لَظَىٰ
16
نَزَّاعَةࣰ لِّلشَّوَىٰ
17
تَدۡعُوا۟ مَنۡ أَدۡبَرَ وَتَوَلَّىٰ
18
وَجَمَعَ فَأَوۡعَىٰۤ
19
۞ إِنَّ ٱلۡإِنسَـٰنَ خُلِقَ هَلُوعًا
20
إِذَا مَسَّهُ ٱلشَّرُّ جَزُوعࣰا
21
وَإِذَا مَسَّهُ ٱلۡخَیۡرُ مَنُوعًا
22
إِلَّا ٱلۡمُصَلِّینَ
23
ٱلَّذِینَ هُمۡ عَلَىٰ صَلَاتِهِمۡ دَاۤىِٕمُونَ
24
وَٱلَّذِینَ فِیۤ أَمۡوَ ٰ⁠لِهِمۡ حَقࣱّ مَّعۡلُومࣱ
25
لِّلسَّاۤىِٕلِ وَٱلۡمَحۡرُومِ
26
وَٱلَّذِینَ یُصَدِّقُونَ بِیَوۡمِ ٱلدِّینِ
27
وَٱلَّذِینَ هُم مِّنۡ عَذَابِ رَبِّهِم مُّشۡفِقُونَ
28
إِنَّ عَذَابَ رَبِّهِمۡ غَیۡرُ مَأۡمُونࣲ
29
وَٱلَّذِینَ هُمۡ لِفُرُوجِهِمۡ حَـٰفِظُونَ
30
إِلَّا عَلَىٰۤ أَزۡوَ ٰ⁠جِهِمۡ أَوۡ مَا مَلَكَتۡ أَیۡمَـٰنُهُمۡ فَإِنَّهُمۡ غَیۡرُ مَلُومِینَ
31
فَمَنِ ٱبۡتَغَىٰ وَرَاۤءَ ذَ ٰ⁠لِكَ فَأُو۟لَـٰۤىِٕكَ هُمُ ٱلۡعَادُونَ
32
وَٱلَّذِینَ هُمۡ لِأَمَـٰنَـٰتِهِمۡ وَعَهۡدِهِمۡ رَ ٰ⁠عُونَ
33
وَٱلَّذِینَ هُم بِشَهَـٰدَ ٰ⁠تِهِمۡ قَاۤىِٕمُونَ
34
وَٱلَّذِینَ هُمۡ عَلَىٰ صَلَاتِهِمۡ یُحَافِظُونَ
35
أُو۟لَـٰۤىِٕكَ فِی جَنَّـٰتࣲ مُّكۡرَمُونَ
36
فَمَالِ ٱلَّذِینَ كَفَرُوا۟ قِبَلَكَ مُهۡطِعِینَ
37
عَنِ ٱلۡیَمِینِ وَعَنِ ٱلشِّمَالِ عِزِینَ
38
أَیَطۡمَعُ كُلُّ ٱمۡرِئࣲ مِّنۡهُمۡ أَن یُدۡخَلَ جَنَّةَ نَعِیمࣲ
39
كَلَّاۤۖ إِنَّا خَلَقۡنَـٰهُم مِّمَّا یَعۡلَمُونَ
40
فَلَاۤ أُقۡسِمُ بِرَبِّ ٱلۡمَشَـٰرِقِ وَٱلۡمَغَـٰرِبِ إِنَّا لَقَـٰدِرُونَ
41
عَلَىٰۤ أَن نُّبَدِّلَ خَیۡرࣰا مِّنۡهُمۡ وَمَا نَحۡنُ بِمَسۡبُوقِینَ
42
فَذَرۡهُمۡ یَخُوضُوا۟ وَیَلۡعَبُوا۟ حَتَّىٰ یُلَـٰقُوا۟ یَوۡمَهُمُ ٱلَّذِی یُوعَدُونَ
43
یَوۡمَ یَخۡرُجُونَ مِنَ ٱلۡأَجۡدَاثِ سِرَاعࣰا كَأَنَّهُمۡ إِلَىٰ نُصُبࣲ یُوفِضُونَ
44
خَـٰشِعَةً أَبۡصَـٰرُهُمۡ تَرۡهَقُهُمۡ ذِلَّةࣱۚ ذَ ٰ⁠لِكَ ٱلۡیَوۡمُ ٱلَّذِی كَانُوا۟ یُوعَدُونَ
