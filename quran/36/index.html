1
بِسۡمِ ٱللَّهِ ٱلرَّحۡمَـٰنِ ٱلرَّحِیمِ یسۤ
2
وَٱلۡقُرۡءَانِ ٱلۡحَكِیمِ
3
إِنَّكَ لَمِنَ ٱلۡمُرۡسَلِینَ
4
عَلَىٰ صِرَ ٰ⁠طࣲ مُّسۡتَقِیمࣲ
5
تَنزِیلَ ٱلۡعَزِیزِ ٱلرَّحِیمِ
6
لِتُنذِرَ قَوۡمࣰا مَّاۤ أُنذِرَ ءَابَاۤؤُهُمۡ فَهُمۡ غَـٰفِلُونَ
7
لَقَدۡ حَقَّ ٱلۡقَوۡلُ عَلَىٰۤ أَكۡثَرِهِمۡ فَهُمۡ لَا یُؤۡمِنُونَ
8
إِنَّا جَعَلۡنَا فِیۤ أَعۡنَـٰقِهِمۡ أَغۡلَـٰلࣰا فَهِیَ إِلَى ٱلۡأَذۡقَانِ فَهُم مُّقۡمَحُونَ
9
وَجَعَلۡنَا مِنۢ بَیۡنِ أَیۡدِیهِمۡ سَدࣰّا وَمِنۡ خَلۡفِهِمۡ سَدࣰّا فَأَغۡشَیۡنَـٰهُمۡ فَهُمۡ لَا یُبۡصِرُونَ
10
وَسَوَاۤءٌ عَلَیۡهِمۡ ءَأَنذَرۡتَهُمۡ أَمۡ لَمۡ تُنذِرۡهُمۡ لَا یُؤۡمِنُونَ
11
إِنَّمَا تُنذِرُ مَنِ ٱتَّبَعَ ٱلذِّكۡرَ وَخَشِیَ ٱلرَّحۡمَـٰنَ بِٱلۡغَیۡبِۖ فَبَشِّرۡهُ بِمَغۡفِرَةࣲ وَأَجۡرࣲ كَرِیمٍ
12
إِنَّا نَحۡنُ نُحۡیِ ٱلۡمَوۡتَىٰ وَنَكۡتُبُ مَا قَدَّمُوا۟ وَءَاثَـٰرَهُمۡۚ وَكُلَّ شَیۡءٍ أَحۡصَیۡنَـٰهُ فِیۤ إِمَامࣲ مُّبِینࣲ
13
وَٱضۡرِبۡ لَهُم مَّثَلًا أَصۡحَـٰبَ ٱلۡقَرۡیَةِ إِذۡ جَاۤءَهَا ٱلۡمُرۡسَلُونَ
14
إِذۡ أَرۡسَلۡنَاۤ إِلَیۡهِمُ ٱثۡنَیۡنِ فَكَذَّبُوهُمَا فَعَزَّزۡنَا بِثَالِثࣲ فَقَالُوۤا۟ إِنَّاۤ إِلَیۡكُم مُّرۡسَلُونَ
15
قَالُوا۟ مَاۤ أَنتُمۡ إِلَّا بَشَرࣱ مِّثۡلُنَا وَمَاۤ أَنزَلَ ٱلرَّحۡمَـٰنُ مِن شَیۡءٍ إِنۡ أَنتُمۡ إِلَّا تَكۡذِبُونَ
16
قَالُوا۟ رَبُّنَا یَعۡلَمُ إِنَّاۤ إِلَیۡكُمۡ لَمُرۡسَلُونَ
17
وَمَا عَلَیۡنَاۤ إِلَّا ٱلۡبَلَـٰغُ ٱلۡمُبِینُ
18
قَالُوۤا۟ إِنَّا تَطَیَّرۡنَا بِكُمۡۖ لَىِٕن لَّمۡ تَنتَهُوا۟ لَنَرۡجُمَنَّكُمۡ وَلَیَمَسَّنَّكُم مِّنَّا عَذَابٌ أَلِیمࣱ
19
قَالُوا۟ طَـٰۤىِٕرُكُم مَّعَكُمۡ أَىِٕن ذُكِّرۡتُمۚ بَلۡ أَنتُمۡ قَوۡمࣱ مُّسۡرِفُونَ
20
وَجَاۤءَ مِنۡ أَقۡصَا ٱلۡمَدِینَةِ رَجُلࣱ یَسۡعَىٰ قَالَ یَـٰقَوۡمِ ٱتَّبِعُوا۟ ٱلۡمُرۡسَلِینَ
21
ٱتَّبِعُوا۟ مَن لَّا یَسۡـَٔلُكُمۡ أَجۡرࣰا وَهُم مُّهۡتَدُونَ
22
وَمَا لِیَ لَاۤ أَعۡبُدُ ٱلَّذِی فَطَرَنِی وَإِلَیۡهِ تُرۡجَعُونَ
23
ءَأَتَّخِذُ مِن دُونِهِۦۤ ءَالِهَةً إِن یُرِدۡنِ ٱلرَّحۡمَـٰنُ بِضُرࣲّ لَّا تُغۡنِ عَنِّی شَفَـٰعَتُهُمۡ شَیۡـࣰٔا وَلَا یُنقِذُونِ
24
إِنِّیۤ إِذࣰا لَّفِی ضَلَـٰلࣲ مُّبِینٍ
25
إِنِّیۤ ءَامَنتُ بِرَبِّكُمۡ فَٱسۡمَعُونِ
26
قِیلَ ٱدۡخُلِ ٱلۡجَنَّةَۖ قَالَ یَـٰلَیۡتَ قَوۡمِی یَعۡلَمُونَ
27
بِمَا غَفَرَ لِی رَبِّی وَجَعَلَنِی مِنَ ٱلۡمُكۡرَمِینَ
28
۞ وَمَاۤ أَنزَلۡنَا عَلَىٰ قَوۡمِهِۦ مِنۢ بَعۡدِهِۦ مِن جُندࣲ مِّنَ ٱلسَّمَاۤءِ وَمَا كُنَّا مُنزِلِینَ
29
إِن كَانَتۡ إِلَّا صَیۡحَةࣰ وَ ٰ⁠حِدَةࣰ فَإِذَا هُمۡ خَـٰمِدُونَ
30
یَـٰحَسۡرَةً عَلَى ٱلۡعِبَادِۚ مَا یَأۡتِیهِم مِّن رَّسُولٍ إِلَّا كَانُوا۟ بِهِۦ یَسۡتَهۡزِءُونَ
31
أَلَمۡ یَرَوۡا۟ كَمۡ أَهۡلَكۡنَا قَبۡلَهُم مِّنَ ٱلۡقُرُونِ أَنَّهُمۡ إِلَیۡهِمۡ لَا یَرۡجِعُونَ
32
وَإِن كُلࣱّ لَّمَّا جَمِیعࣱ لَّدَیۡنَا مُحۡضَرُونَ
33
وَءَایَةࣱ لَّهُمُ ٱلۡأَرۡضُ ٱلۡمَیۡتَةُ أَحۡیَیۡنَـٰهَا وَأَخۡرَجۡنَا مِنۡهَا حَبࣰّا فَمِنۡهُ یَأۡكُلُونَ
34
وَجَعَلۡنَا فِیهَا جَنَّـٰتࣲ مِّن نَّخِیلࣲ وَأَعۡنَـٰبࣲ وَفَجَّرۡنَا فِیهَا مِنَ ٱلۡعُیُونِ
35
لِیَأۡكُلُوا۟ مِن ثَمَرِهِۦ وَمَا عَمِلَتۡهُ أَیۡدِیهِمۡۚ أَفَلَا یَشۡكُرُونَ
36
سُبۡحَـٰنَ ٱلَّذِی خَلَقَ ٱلۡأَزۡوَ ٰ⁠جَ كُلَّهَا مِمَّا تُنۢبِتُ ٱلۡأَرۡضُ وَمِنۡ أَنفُسِهِمۡ وَمِمَّا لَا یَعۡلَمُونَ
37
وَءَایَةࣱ لَّهُمُ ٱلَّیۡلُ نَسۡلَخُ مِنۡهُ ٱلنَّهَارَ فَإِذَا هُم مُّظۡلِمُونَ
38
وَٱلشَّمۡسُ تَجۡرِی لِمُسۡتَقَرࣲّ لَّهَاۚ ذَ ٰ⁠لِكَ تَقۡدِیرُ ٱلۡعَزِیزِ ٱلۡعَلِیمِ
39
وَٱلۡقَمَرَ قَدَّرۡنَـٰهُ مَنَازِلَ حَتَّىٰ عَادَ كَٱلۡعُرۡجُونِ ٱلۡقَدِیمِ
40
لَا ٱلشَّمۡسُ یَنۢبَغِی لَهَاۤ أَن تُدۡرِكَ ٱلۡقَمَرَ وَلَا ٱلَّیۡلُ سَابِقُ ٱلنَّهَارِۚ وَكُلࣱّ فِی فَلَكࣲ یَسۡبَحُونَ
41
وَءَایَةࣱ لَّهُمۡ أَنَّا حَمَلۡنَا ذُرِّیَّتَهُمۡ فِی ٱلۡفُلۡكِ ٱلۡمَشۡحُونِ
42
وَخَلَقۡنَا لَهُم مِّن مِّثۡلِهِۦ مَا یَرۡكَبُونَ
43
وَإِن نَّشَأۡ نُغۡرِقۡهُمۡ فَلَا صَرِیخَ لَهُمۡ وَلَا هُمۡ یُنقَذُونَ
44
إِلَّا رَحۡمَةࣰ مِّنَّا وَمَتَـٰعًا إِلَىٰ حِینࣲ
45
وَإِذَا قِیلَ لَهُمُ ٱتَّقُوا۟ مَا بَیۡنَ أَیۡدِیكُمۡ وَمَا خَلۡفَكُمۡ لَعَلَّكُمۡ تُرۡحَمُونَ
46
وَمَا تَأۡتِیهِم مِّنۡ ءَایَةࣲ مِّنۡ ءَایَـٰتِ رَبِّهِمۡ إِلَّا كَانُوا۟ عَنۡهَا مُعۡرِضِینَ
47
وَإِذَا قِیلَ لَهُمۡ أَنفِقُوا۟ مِمَّا رَزَقَكُمُ ٱللَّهُ قَالَ ٱلَّذِینَ كَفَرُوا۟ لِلَّذِینَ ءَامَنُوۤا۟ أَنُطۡعِمُ مَن لَّوۡ یَشَاۤءُ ٱللَّهُ أَطۡعَمَهُۥۤ إِنۡ أَنتُمۡ إِلَّا فِی ضَلَـٰلࣲ مُّبِینࣲ
48
وَیَقُولُونَ مَتَىٰ هَـٰذَا ٱلۡوَعۡدُ إِن كُنتُمۡ صَـٰدِقِینَ
49
مَا یَنظُرُونَ إِلَّا صَیۡحَةࣰ وَ ٰ⁠حِدَةࣰ تَأۡخُذُهُمۡ وَهُمۡ یَخِصِّمُونَ
50
فَلَا یَسۡتَطِیعُونَ تَوۡصِیَةࣰ وَلَاۤ إِلَىٰۤ أَهۡلِهِمۡ یَرۡجِعُونَ
51
وَنُفِخَ فِی ٱلصُّورِ فَإِذَا هُم مِّنَ ٱلۡأَجۡدَاثِ إِلَىٰ رَبِّهِمۡ یَنسِلُونَ
52
قَالُوا۟ یَـٰوَیۡلَنَا مَنۢ بَعَثَنَا مِن مَّرۡقَدِنَاۜۗ هَـٰذَا مَا وَعَدَ ٱلرَّحۡمَـٰنُ وَصَدَقَ ٱلۡمُرۡسَلُونَ
53
إِن كَانَتۡ إِلَّا صَیۡحَةࣰ وَ ٰ⁠حِدَةࣰ فَإِذَا هُمۡ جَمِیعࣱ لَّدَیۡنَا مُحۡضَرُونَ
54
فَٱلۡیَوۡمَ لَا تُظۡلَمُ نَفۡسࣱ شَیۡـࣰٔا وَلَا تُجۡزَوۡنَ إِلَّا مَا كُنتُمۡ تَعۡمَلُونَ
55
إِنَّ أَصۡحَـٰبَ ٱلۡجَنَّةِ ٱلۡیَوۡمَ فِی شُغُلࣲ فَـٰكِهُونَ
56
هُمۡ وَأَزۡوَ ٰ⁠جُهُمۡ فِی ظِلَـٰلٍ عَلَى ٱلۡأَرَاۤىِٕكِ مُتَّكِـُٔونَ
57
لَهُمۡ فِیهَا فَـٰكِهَةࣱ وَلَهُم مَّا یَدَّعُونَ
58
سَلَـٰمࣱ قَوۡلࣰا مِّن رَّبࣲّ رَّحِیمࣲ
59
وَٱمۡتَـٰزُوا۟ ٱلۡیَوۡمَ أَیُّهَا ٱلۡمُجۡرِمُونَ
60
۞ أَلَمۡ أَعۡهَدۡ إِلَیۡكُمۡ یَـٰبَنِیۤ ءَادَمَ أَن لَّا تَعۡبُدُوا۟ ٱلشَّیۡطَـٰنَۖ إِنَّهُۥ لَكُمۡ عَدُوࣱّ مُّبِینࣱ
61
وَأَنِ ٱعۡبُدُونِیۚ هَـٰذَا صِرَ ٰ⁠طࣱ مُّسۡتَقِیمࣱ
62
وَلَقَدۡ أَضَلَّ مِنكُمۡ جِبِلࣰّا كَثِیرًاۖ أَفَلَمۡ تَكُونُوا۟ تَعۡقِلُونَ
63
هَـٰذِهِۦ جَهَنَّمُ ٱلَّتِی كُنتُمۡ تُوعَدُونَ
64
ٱصۡلَوۡهَا ٱلۡیَوۡمَ بِمَا كُنتُمۡ تَكۡفُرُونَ
65
ٱلۡیَوۡمَ نَخۡتِمُ عَلَىٰۤ أَفۡوَ ٰ⁠هِهِمۡ وَتُكَلِّمُنَاۤ أَیۡدِیهِمۡ وَتَشۡهَدُ أَرۡجُلُهُم بِمَا كَانُوا۟ یَكۡسِبُونَ
66
وَلَوۡ نَشَاۤءُ لَطَمَسۡنَا عَلَىٰۤ أَعۡیُنِهِمۡ فَٱسۡتَبَقُوا۟ ٱلصِّرَ ٰ⁠طَ فَأَنَّىٰ یُبۡصِرُونَ
67
وَلَوۡ نَشَاۤءُ لَمَسَخۡنَـٰهُمۡ عَلَىٰ مَكَانَتِهِمۡ فَمَا ٱسۡتَطَـٰعُوا۟ مُضِیࣰّا وَلَا یَرۡجِعُونَ
68
وَمَن نُّعَمِّرۡهُ نُنَكِّسۡهُ فِی ٱلۡخَلۡقِۚ أَفَلَا یَعۡقِلُونَ
69
وَمَا عَلَّمۡنَـٰهُ ٱلشِّعۡرَ وَمَا یَنۢبَغِی لَهُۥۤۚ إِنۡ هُوَ إِلَّا ذِكۡرࣱ وَقُرۡءَانࣱ مُّبِینࣱ
70
لِّیُنذِرَ مَن كَانَ حَیࣰّا وَیَحِقَّ ٱلۡقَوۡلُ عَلَى ٱلۡكَـٰفِرِینَ
71
أَوَلَمۡ یَرَوۡا۟ أَنَّا خَلَقۡنَا لَهُم مِّمَّا عَمِلَتۡ أَیۡدِینَاۤ أَنۡعَـٰمࣰا فَهُمۡ لَهَا مَـٰلِكُونَ
72
وَذَلَّلۡنَـٰهَا لَهُمۡ فَمِنۡهَا رَكُوبُهُمۡ وَمِنۡهَا یَأۡكُلُونَ
73
وَلَهُمۡ فِیهَا مَنَـٰفِعُ وَمَشَارِبُۚ أَفَلَا یَشۡكُرُونَ
74
وَٱتَّخَذُوا۟ مِن دُونِ ٱللَّهِ ءَالِهَةࣰ لَّعَلَّهُمۡ یُنصَرُونَ
75
لَا یَسۡتَطِیعُونَ نَصۡرَهُمۡ وَهُمۡ لَهُمۡ جُندࣱ مُّحۡضَرُونَ
76
فَلَا یَحۡزُنكَ قَوۡلُهُمۡۘ إِنَّا نَعۡلَمُ مَا یُسِرُّونَ وَمَا یُعۡلِنُونَ
77
أَوَلَمۡ یَرَ ٱلۡإِنسَـٰنُ أَنَّا خَلَقۡنَـٰهُ مِن نُّطۡفَةࣲ فَإِذَا هُوَ خَصِیمࣱ مُّبِینࣱ
78
وَضَرَبَ لَنَا مَثَلࣰا وَنَسِیَ خَلۡقَهُۥۖ قَالَ مَن یُحۡیِ ٱلۡعِظَـٰمَ وَهِیَ رَمِیمࣱ
79
قُلۡ یُحۡیِیهَا ٱلَّذِیۤ أَنشَأَهَاۤ أَوَّلَ مَرَّةࣲۖ وَهُوَ بِكُلِّ خَلۡقٍ عَلِیمٌ
80
ٱلَّذِی جَعَلَ لَكُم مِّنَ ٱلشَّجَرِ ٱلۡأَخۡضَرِ نَارࣰا فَإِذَاۤ أَنتُم مِّنۡهُ تُوقِدُونَ
81
