تنزيل
1
بِسۡمِ ٱللَّهِ ٱلرَّحۡمَـٰنِ ٱلرَّحِیمِ إِذَا وَقَعَتِ ٱلۡوَاقِعَةُ
2
لَیۡسَ لِوَقۡعَتِهَا كَاذِبَةٌ
3
خَافِضَةࣱ رَّافِعَةٌ
4
إِذَا رُجَّتِ ٱلۡأَرۡضُ رَجࣰّا
5
وَبُسَّتِ ٱلۡجِبَالُ بَسࣰّا
6
فَكَانَتۡ هَبَاۤءࣰ مُّنۢبَثࣰّا
7
وَكُنتُمۡ أَزۡوَ ٰ⁠جࣰا ثَلَـٰثَةࣰ
8
فَأَصۡحَـٰبُ ٱلۡمَیۡمَنَةِ مَاۤ أَصۡحَـٰبُ ٱلۡمَیۡمَنَةِ
9
وَأَصۡحَـٰبُ ٱلۡمَشۡـَٔمَةِ مَاۤ أَصۡحَـٰبُ ٱلۡمَشۡـَٔمَةِ
10
وَٱلسَّـٰبِقُونَ ٱلسَّـٰبِقُونَ
11
أُو۟لَـٰۤىِٕكَ ٱلۡمُقَرَّبُونَ
12
فِی جَنَّـٰتِ ٱلنَّعِیمِ
13
ثُلَّةࣱ مِّنَ ٱلۡأَوَّلِینَ
14
وَقَلِیلࣱ مِّنَ ٱلۡـَٔاخِرِینَ
15
عَلَىٰ سُرُرࣲ مَّوۡضُونَةࣲ
16
مُّتَّكِـِٔینَ عَلَیۡهَا مُتَقَـٰبِلِینَ
17
یَطُوفُ عَلَیۡهِمۡ وِلۡدَ ٰ⁠نࣱ مُّخَلَّدُونَ
18
بِأَكۡوَابࣲ وَأَبَارِیقَ وَكَأۡسࣲ مِّن مَّعِینࣲ
19
لَّا یُصَدَّعُونَ عَنۡهَا وَلَا یُنزِفُونَ
20
وَفَـٰكِهَةࣲ مِّمَّا یَتَخَیَّرُونَ
21
وَلَحۡمِ طَیۡرࣲ مِّمَّا یَشۡتَهُونَ
22
وَحُورٌ عِینࣱ
23
كَأَمۡثَـٰلِ ٱللُّؤۡلُوِٕ ٱلۡمَكۡنُونِ
24
جَزَاۤءَۢ بِمَا كَانُوا۟ یَعۡمَلُونَ
25
لَا یَسۡمَعُونَ فِیهَا لَغۡوࣰا وَلَا تَأۡثِیمًا
26
إِلَّا قِیلࣰا سَلَـٰمࣰا سَلَـٰمࣰا
27
وَأَصۡحَـٰبُ ٱلۡیَمِینِ مَاۤ أَصۡحَـٰبُ ٱلۡیَمِینِ
28
فِی سِدۡرࣲ مَّخۡضُودࣲ
29
وَطَلۡحࣲ مَّنضُودࣲ
30
وَظِلࣲّ مَّمۡدُودࣲ
31
وَمَاۤءࣲ مَّسۡكُوبࣲ
32
وَفَـٰكِهَةࣲ كَثِیرَةࣲ
33
لَّا مَقۡطُوعَةࣲ وَلَا مَمۡنُوعَةࣲ
34
وَفُرُشࣲ مَّرۡفُوعَةٍ
35
إِنَّاۤ أَنشَأۡنَـٰهُنَّ إِنشَاۤءࣰ
36
فَجَعَلۡنَـٰهُنَّ أَبۡكَارًا
37
عُرُبًا أَتۡرَابࣰا
38
لِّأَصۡحَـٰبِ ٱلۡیَمِینِ
39
ثُلَّةࣱ مِّنَ ٱلۡأَوَّلِینَ
40
وَثُلَّةࣱ مِّنَ ٱلۡـَٔاخِرِینَ
41
وَأَصۡحَـٰبُ ٱلشِّمَالِ مَاۤ أَصۡحَـٰبُ ٱلشِّمَالِ
42
فِی سَمُومࣲ وَحَمِیمࣲ
43
وَظِلࣲّ مِّن یَحۡمُومࣲ
44
لَّا بَارِدࣲ وَلَا كَرِیمٍ
45
إِنَّهُمۡ كَانُوا۟ قَبۡلَ ذَ ٰ⁠لِكَ مُتۡرَفِینَ
46
وَكَانُوا۟ یُصِرُّونَ عَلَى ٱلۡحِنثِ ٱلۡعَظِیمِ
47
وَكَانُوا۟ یَقُولُونَ أَىِٕذَا مِتۡنَا وَكُنَّا تُرَابࣰا وَعِظَـٰمًا أَءِنَّا لَمَبۡعُوثُونَ
48
أَوَءَابَاۤؤُنَا ٱلۡأَوَّلُونَ
49
قُلۡ إِنَّ ٱلۡأَوَّلِینَ وَٱلۡـَٔاخِرِینَ
50
لَمَجۡمُوعُونَ إِلَىٰ مِیقَـٰتِ یَوۡمࣲ مَّعۡلُومࣲ
51
ثُمَّ إِنَّكُمۡ أَیُّهَا ٱلضَّاۤلُّونَ ٱلۡمُكَذِّبُونَ
52
لَـَٔاكِلُونَ مِن شَجَرࣲ مِّن زَقُّومࣲ
53
فَمَالِـُٔونَ مِنۡهَا ٱلۡبُطُونَ
54
فَشَـٰرِبُونَ عَلَیۡهِ مِنَ ٱلۡحَمِیمِ
55
فَشَـٰرِبُونَ شُرۡبَ ٱلۡهِیمِ
56
هَـٰذَا نُزُلُهُمۡ یَوۡمَ ٱلدِّینِ
57
نَحۡنُ خَلَقۡنَـٰكُمۡ فَلَوۡلَا تُصَدِّقُونَ
58
أَفَرَءَیۡتُم مَّا تُمۡنُونَ
59
ءَأَنتُمۡ تَخۡلُقُونَهُۥۤ أَمۡ نَحۡنُ ٱلۡخَـٰلِقُونَ
60
نَحۡنُ قَدَّرۡنَا بَیۡنَكُمُ ٱلۡمَوۡتَ وَمَا نَحۡنُ بِمَسۡبُوقِینَ
61
عَلَىٰۤ أَن نُّبَدِّلَ أَمۡثَـٰلَكُمۡ وَنُنشِئَكُمۡ فِی مَا لَا تَعۡلَمُونَ
62
وَلَقَدۡ عَلِمۡتُمُ ٱلنَّشۡأَةَ ٱلۡأُولَىٰ فَلَوۡلَا تَذَكَّرُونَ
63
أَفَرَءَیۡتُم مَّا تَحۡرُثُونَ
64
ءَأَنتُمۡ تَزۡرَعُونَهُۥۤ أَمۡ نَحۡنُ ٱلزَّ ٰ⁠رِعُونَ
65
لَوۡ نَشَاۤءُ لَجَعَلۡنَـٰهُ حُطَـٰمࣰا فَظَلۡتُمۡ تَفَكَّهُونَ
66
إِنَّا لَمُغۡرَمُونَ
67
بَلۡ نَحۡنُ مَحۡرُومُونَ
68
أَفَرَءَیۡتُمُ ٱلۡمَاۤءَ ٱلَّذِی تَشۡرَبُونَ
69
ءَأَنتُمۡ أَنزَلۡتُمُوهُ مِنَ ٱلۡمُزۡنِ أَمۡ نَحۡنُ ٱلۡمُنزِلُونَ
70
لَوۡ نَشَاۤءُ جَعَلۡنَـٰهُ أُجَاجࣰا فَلَوۡلَا تَشۡكُرُونَ
71
أَفَرَءَیۡتُمُ ٱلنَّارَ ٱلَّتِی تُورُونَ
72
ءَأَنتُمۡ أَنشَأۡتُمۡ شَجَرَتَهَاۤ أَمۡ نَحۡنُ ٱلۡمُنشِـُٔونَ
73
نَحۡنُ جَعَلۡنَـٰهَا تَذۡكِرَةࣰ وَمَتَـٰعࣰا لِّلۡمُقۡوِینَ
74
فَسَبِّحۡ بِٱسۡمِ رَبِّكَ ٱلۡعَظِیمِ
75
۞ فَلَاۤ أُقۡسِمُ بِمَوَ ٰ⁠قِعِ ٱلنُّجُومِ
76
وَإِنَّهُۥ لَقَسَمࣱ لَّوۡ تَعۡلَمُونَ عَظِیمٌ
77
إِنَّهُۥ لَقُرۡءَانࣱ كَرِیمࣱ
78
فِی كِتَـٰبࣲ مَّكۡنُونࣲ
79
لَّا یَمَسُّهُۥۤ إِلَّا ٱلۡمُطَهَّرُونَ
80
تَنزِیلࣱ مِّن رَّبِّ ٱلۡعَـٰلَمِینَ
81
أَفَبِهَـٰذَا ٱلۡحَدِیثِ أَنتُم مُّدۡهِنُونَ
82
وَتَجۡعَلُونَ رِزۡقَكُمۡ أَنَّكُمۡ تُكَذِّبُونَ
83
فَلَوۡلَاۤ إِذَا بَلَغَتِ ٱلۡحُلۡقُومَ
84
وَأَنتُمۡ حِینَىِٕذࣲ تَنظُرُونَ
85
وَنَحۡنُ أَقۡرَبُ إِلَیۡهِ مِنكُمۡ وَلَـٰكِن لَّا تُبۡصِرُونَ
86
فَلَوۡلَاۤ إِن كُنتُمۡ غَیۡرَ مَدِینِینَ
87
تَرۡجِعُونَهَاۤ إِن كُنتُمۡ صَـٰدِقِینَ
88
فَأَمَّاۤ إِن كَانَ مِنَ ٱلۡمُقَرَّبِینَ
89
فَرَوۡحࣱ وَرَیۡحَانࣱ وَجَنَّتُ نَعِیمࣲ
90
وَأَمَّاۤ إِن كَانَ مِنۡ أَصۡحَـٰبِ ٱلۡیَمِینِ
91
فَسَلَـٰمࣱ لَّكَ مِنۡ أَصۡحَـٰبِ ٱلۡیَمِینِ
92
وَأَمَّاۤ إِن كَانَ مِنَ ٱلۡمُكَذِّبِینَ ٱلضَّاۤلِّینَ
93
فَنُزُلࣱ مِّنۡ حَمِیمࣲ
94
وَتَصۡلِیَةُ جَحِیمٍ
95
إِنَّ هَـٰذَا لَهُوَ حَقُّ ٱلۡیَقِینِ
96
فَسَبِّحۡ بِٱسۡمِ رَبِّكَ ٱلۡعَظِیمِ
