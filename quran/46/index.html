سُورَةُ الأَحۡقَافِ
Al-Ahqaf — The Dunes
مكية · 35 آية
الآية 1
بِسۡمِ ٱللَّهِ ٱلرَّحۡمَـٰنِ ٱلرَّحِیمِ حمۤ
الآية 2
تَنزِیلُ ٱلۡكِتَـٰبِ مِنَ ٱللَّهِ ٱلۡعَزِیزِ ٱلۡحَكِیمِ
الآية 3
مَا خَلَقۡنَا ٱلسَّمَـٰوَ ٰ⁠تِ وَٱلۡأَرۡضَ وَمَا بَیۡنَهُمَاۤ إِلَّا بِٱلۡحَقِّ وَأَجَلࣲ مُّسَمࣰّىۚ وَٱلَّذِینَ كَفَرُوا۟ عَمَّاۤ أُنذِرُوا۟ مُعۡرِضُونَ
الآية 4
قُلۡ أَرَءَیۡتُم مَّا تَدۡعُونَ مِن دُونِ ٱللَّهِ أَرُونِی مَاذَا خَلَقُوا۟ مِنَ ٱلۡأَرۡضِ أَمۡ لَهُمۡ شِرۡكࣱ فِی ٱلسَّمَـٰوَ ٰ⁠تِۖ ٱئۡتُونِی بِكِتَـٰبࣲ مِّن قَبۡلِ هَـٰذَاۤ أَوۡ أَثَـٰرَةࣲ مِّنۡ عِلۡمٍ إِن كُنتُمۡ صَـٰدِقِینَ
الآية 5
وَمَنۡ أَضَلُّ مِمَّن یَدۡعُوا۟ مِن دُونِ ٱللَّهِ مَن لَّا یَسۡتَجِیبُ لَهُۥۤ إِلَىٰ یَوۡمِ ٱلۡقِیَـٰمَةِ وَهُمۡ عَن دُعَاۤىِٕهِمۡ غَـٰفِلُونَ
الآية 6
وَإِذَا حُشِرَ ٱلنَّاسُ كَانُوا۟ لَهُمۡ أَعۡدَاۤءࣰ وَكَانُوا۟ بِعِبَادَتِهِمۡ كَـٰفِرِینَ
الآية 7
وَإِذَا تُتۡلَىٰ عَلَیۡهِمۡ ءَایَـٰتُنَا بَیِّنَـٰتࣲ قَالَ ٱلَّذِینَ كَفَرُوا۟ لِلۡحَقِّ لَمَّا جَاۤءَهُمۡ هَـٰذَا سِحۡرࣱ مُّبِینٌ
الآية 8
أَمۡ یَقُولُونَ ٱفۡتَرَىٰهُۖ قُلۡ إِنِ ٱفۡتَرَیۡتُهُۥ فَلَا تَمۡلِكُونَ لِی مِنَ ٱللَّهِ شَیۡـًٔاۖ هُوَ أَعۡلَمُ بِمَا تُفِیضُونَ فِیهِۚ كَفَىٰ بِهِۦ شَهِیدَۢا بَیۡنِی وَبَیۡنَكُمۡۖ وَهُوَ ٱلۡغَفُورُ ٱلرَّحِیمُ
الآية 9
قُلۡ مَا كُنتُ بِدۡعࣰا مِّنَ ٱلرُّسُلِ وَمَاۤ أَدۡرِی مَا یُفۡعَلُ بِی وَلَا بِكُمۡۖ إِنۡ أَتَّبِعُ إِلَّا مَا یُوحَىٰۤ إِلَیَّ وَمَاۤ أَنَا۠ إِلَّا نَذِیرࣱ مُّبِینࣱ
الآية 10
قُلۡ أَرَءَیۡتُمۡ إِن كَانَ مِنۡ عِندِ ٱللَّهِ وَكَفَرۡتُم بِهِۦ وَشَهِدَ شَاهِدࣱ مِّنۢ بَنِیۤ إِسۡرَ ٰ⁠ۤءِیلَ عَلَىٰ مِثۡلِهِۦ فَـَٔامَنَ وَٱسۡتَكۡبَرۡتُمۡۚ إِنَّ ٱللَّهَ لَا یَهۡدِی ٱلۡقَوۡمَ ٱلظَّـٰلِمِینَ
الآية 11
وَقَالَ ٱلَّذِینَ كَفَرُوا۟ لِلَّذِینَ ءَامَنُوا۟ لَوۡ كَانَ خَیۡرࣰا مَّا سَبَقُونَاۤ إِلَیۡهِۚ وَإِذۡ لَمۡ یَهۡتَدُوا۟ بِهِۦ فَسَیَقُولُونَ هَـٰذَاۤ إِفۡكࣱ قَدِیمࣱ
الآية 12
وَمِن قَبۡلِهِۦ كِتَـٰبُ مُوسَىٰۤ إِمَامࣰا وَرَحۡمَةࣰۚ وَهَـٰذَا كِتَـٰبࣱ مُّصَدِّقࣱ لِّسَانًا عَرَبِیࣰّا لِّیُنذِرَ ٱلَّذِینَ ظَلَمُوا۟ وَبُشۡرَىٰ لِلۡمُحۡسِنِینَ
الآية 13
إِنَّ ٱلَّذِینَ قَالُوا۟ رَبُّنَا ٱللَّهُ ثُمَّ ٱسۡتَقَـٰمُوا۟ فَلَا خَوۡفٌ عَلَیۡهِمۡ وَلَا هُمۡ یَحۡزَنُونَ
الآية 14
أُو۟لَـٰۤىِٕكَ أَصۡحَـٰبُ ٱلۡجَنَّةِ خَـٰلِدِینَ فِیهَا جَزَاۤءَۢ بِمَا كَانُوا۟ یَعۡمَلُونَ
الآية 15
وَوَصَّیۡنَا ٱلۡإِنسَـٰنَ بِوَ ٰ⁠لِدَیۡهِ إِحۡسَـٰنًاۖ حَمَلَتۡهُ أُمُّهُۥ كُرۡهࣰا وَوَضَعَتۡهُ كُرۡهࣰاۖ وَحَمۡلُهُۥ وَفِصَـٰلُهُۥ ثَلَـٰثُونَ شَهۡرًاۚ حَتَّىٰۤ إِذَا بَلَغَ أَشُدَّهُۥ وَبَلَغَ أَرۡبَعِینَ سَنَةࣰ قَالَ رَبِّ أَوۡزِعۡنِیۤ أَنۡ أَشۡكُرَ نِعۡمَتَكَ ٱلَّتِیۤ أَنۡعَمۡتَ عَلَیَّ وَعَلَىٰ وَ ٰ⁠لِدَیَّ وَأَنۡ أَعۡمَلَ صَـٰلِحࣰا تَرۡضَىٰهُ وَأَصۡلِحۡ لِی فِی ذُرِّیَّتِیۤۖ إِنِّی تُبۡتُ إِلَیۡكَ وَإِنِّی مِنَ ٱلۡمُسۡلِمِینَ
الآية 16
أُو۟لَـٰۤىِٕكَ ٱلَّذِینَ نَتَقَبَّلُ عَنۡهُمۡ أَحۡسَنَ مَا عَمِلُوا۟ وَنَتَجَاوَزُ عَن سَیِّـَٔاتِهِمۡ فِیۤ أَصۡحَـٰبِ ٱلۡجَنَّةِۖ وَعۡدَ ٱلصِّدۡقِ ٱلَّذِی كَانُوا۟ یُوعَدُونَ
الآية 17
وَٱلَّذِی قَالَ لِوَ ٰ⁠لِدَیۡهِ أُفࣲّ لَّكُمَاۤ أَتَعِدَانِنِیۤ أَنۡ أُخۡرَجَ وَقَدۡ خَلَتِ ٱلۡقُرُونُ مِن قَبۡلِی وَهُمَا یَسۡتَغِیثَانِ ٱللَّهَ وَیۡلَكَ ءَامِنۡ إِنَّ وَعۡدَ ٱللَّهِ حَقࣱّ فَیَقُولُ مَا هَـٰذَاۤ إِلَّاۤ أَسَـٰطِیرُ ٱلۡأَوَّلِینَ
الآية 18
أُو۟لَـٰۤىِٕكَ ٱلَّذِینَ حَقَّ عَلَیۡهِمُ ٱلۡقَوۡلُ فِیۤ أُمَمࣲ قَدۡ خَلَتۡ مِن قَبۡلِهِم مِّنَ ٱلۡجِنِّ وَٱلۡإِنسِۖ إِنَّهُمۡ كَانُوا۟ خَـٰسِرِینَ
الآية 19
وَلِكُلࣲّ دَرَجَـٰتࣱ مِّمَّا عَمِلُوا۟ۖ وَلِیُوَفِّیَهُمۡ أَعۡمَـٰلَهُمۡ وَهُمۡ لَا یُظۡلَمُونَ
الآية 20
وَیَوۡمَ یُعۡرَضُ ٱلَّذِینَ كَفَرُوا۟ عَلَى ٱلنَّارِ أَذۡهَبۡتُمۡ طَیِّبَـٰتِكُمۡ فِی حَیَاتِكُمُ ٱلدُّنۡیَا وَٱسۡتَمۡتَعۡتُم بِهَا فَٱلۡیَوۡمَ تُجۡزَوۡنَ عَذَابَ ٱلۡهُونِ بِمَا كُنتُمۡ تَسۡتَكۡبِرُونَ فِی ٱلۡأَرۡضِ بِغَیۡرِ ٱلۡحَقِّ وَبِمَا كُنتُمۡ تَفۡسُقُونَ
الآية 21
۞ وَٱذۡكُرۡ أَخَا عَادٍ إِذۡ أَنذَرَ قَوۡمَهُۥ بِٱلۡأَحۡقَافِ وَقَدۡ خَلَتِ ٱلنُّذُرُ مِنۢ بَیۡنِ یَدَیۡهِ وَمِنۡ خَلۡفِهِۦۤ أَلَّا تَعۡبُدُوۤا۟ إِلَّا ٱللَّهَ إِنِّیۤ أَخَافُ عَلَیۡكُمۡ عَذَابَ یَوۡمٍ عَظِیمࣲ
الآية 22
قَالُوۤا۟ أَجِئۡتَنَا لِتَأۡفِكَنَا عَنۡ ءَالِهَتِنَا فَأۡتِنَا بِمَا تَعِدُنَاۤ إِن كُنتَ مِنَ ٱلصَّـٰدِقِینَ
الآية 23
قَالَ إِنَّمَا ٱلۡعِلۡمُ عِندَ ٱللَّهِ وَأُبَلِّغُكُم مَّاۤ أُرۡسِلۡتُ بِهِۦ وَلَـٰكِنِّیۤ أَرَىٰكُمۡ قَوۡمࣰا تَجۡهَلُونَ
الآية 24
فَلَمَّا رَأَوۡهُ عَارِضࣰا مُّسۡتَقۡبِلَ أَوۡدِیَتِهِمۡ قَالُوا۟ هَـٰذَا عَارِضࣱ مُّمۡطِرُنَاۚ بَلۡ هُوَ مَا ٱسۡتَعۡجَلۡتُم بِهِۦۖ رِیحࣱ فِیهَا عَذَابٌ أَلِیمࣱ
الآية 25
تُدَمِّرُ كُلَّ شَیۡءِۭ بِأَمۡرِ رَبِّهَا فَأَصۡبَحُوا۟ لَا یُرَىٰۤ إِلَّا مَسَـٰكِنُهُمۡۚ كَذَ ٰ⁠لِكَ نَجۡزِی ٱلۡقَوۡمَ ٱلۡمُجۡرِمِینَ
الآية 26
وَلَقَدۡ مَكَّنَّـٰهُمۡ فِیمَاۤ إِن مَّكَّنَّـٰكُمۡ فِیهِ وَجَعَلۡنَا لَهُمۡ سَمۡعࣰا وَأَبۡصَـٰرࣰا وَأَفۡـِٔدَةࣰ فَمَاۤ أَغۡنَىٰ عَنۡهُمۡ سَمۡعُهُمۡ وَلَاۤ أَبۡصَـٰرُهُمۡ وَلَاۤ أَفۡـِٔدَتُهُم مِّن شَیۡءٍ إِذۡ كَانُوا۟ یَجۡحَدُونَ بِـَٔایَـٰتِ ٱللَّهِ وَحَاقَ بِهِم مَّا كَانُوا۟ بِهِۦ یَسۡتَهۡزِءُونَ
الآية 27
وَلَقَدۡ أَهۡلَكۡنَا مَا حَوۡلَكُم مِّنَ ٱلۡقُرَىٰ وَصَرَّفۡنَا ٱلۡـَٔایَـٰتِ لَعَلَّهُمۡ یَرۡجِعُونَ
الآية 28
فَلَوۡلَا نَصَرَهُمُ ٱلَّذِینَ ٱتَّخَذُوا۟ مِن دُونِ ٱللَّهِ قُرۡبَانًا ءَالِهَةَۢۖ بَلۡ ضَلُّوا۟ عَنۡهُمۡۚ وَذَ ٰ⁠لِكَ إِفۡكُهُمۡ وَمَا كَانُوا۟ یَفۡتَرُونَ
الآية 29
وَإِذۡ صَرَفۡنَاۤ إِلَیۡكَ نَفَرࣰا مِّنَ ٱلۡجِنِّ یَسۡتَمِعُونَ ٱلۡقُرۡءَانَ فَلَمَّا حَضَرُوهُ قَالُوۤا۟ أَنصِتُوا۟ۖ فَلَمَّا قُضِیَ وَلَّوۡا۟ إِلَىٰ قَوۡمِهِم مُّنذِرِینَ
الآية 30
قَالُوا۟ یَـٰقَوۡمَنَاۤ إِنَّا سَمِعۡنَا كِتَـٰبًا أُنزِلَ مِنۢ بَعۡدِ مُوسَىٰ مُصَدِّقࣰا لِّمَا بَیۡنَ یَدَیۡهِ یَهۡدِیۤ إِلَى ٱلۡحَقِّ وَإِلَىٰ طَرِیقࣲ مُّسۡتَقِیمࣲ
الآية 31
یَـٰقَوۡمَنَاۤ أَجِیبُوا۟ دَاعِیَ ٱللَّهِ وَءَامِنُوا۟ بِهِۦ یَغۡفِرۡ لَكُم مِّن ذُنُوبِكُمۡ وَیُجِرۡكُم مِّنۡ عَذَابٍ أَلِیمࣲ
الآية 32
وَمَن لَّا یُجِبۡ دَاعِیَ ٱللَّهِ فَلَیۡسَ بِمُعۡجِزࣲ فِی ٱلۡأَرۡضِ وَلَیۡسَ لَهُۥ مِن دُونِهِۦۤ أَوۡلِیَاۤءُۚ أُو۟لَـٰۤىِٕكَ فِی ضَلَـٰلࣲ مُّبِینٍ
الآية 33
أَوَلَمۡ یَرَوۡا۟ أَنَّ ٱللَّهَ ٱلَّذِی خَلَقَ ٱلسَّمَـٰوَ ٰ⁠تِ وَٱلۡأَرۡضَ وَلَمۡ یَعۡیَ بِخَلۡقِهِنَّ بِقَـٰدِرٍ عَلَىٰۤ أَن یُحۡـِۧیَ ٱلۡمَوۡتَىٰۚ بَلَىٰۤۚ إِنَّهُۥ عَلَىٰ كُلِّ شَیۡءࣲ قَدِیرࣱ
الآية 34
وَیَوۡمَ یُعۡرَضُ ٱلَّذِینَ كَفَرُوا۟ عَلَى ٱلنَّارِ أَلَیۡسَ هَـٰذَا بِٱلۡحَقِّۖ قَالُوا۟ بَلَىٰ وَرَبِّنَاۚ قَالَ فَذُوقُوا۟ ٱلۡعَذَابَ بِمَا كُنتُمۡ تَكۡفُرُونَ
الآية 35
فَٱصۡبِرۡ كَمَا صَبَرَ أُو۟لُوا۟ ٱلۡعَزۡمِ مِنَ ٱلرُّسُلِ وَلَا تَسۡتَعۡجِل لَّهُمۡۚ كَأَنَّهُمۡ یَوۡمَ یَرَوۡنَ مَا یُوعَدُونَ لَمۡ یَلۡبَثُوۤا۟ إِلَّا سَاعَةࣰ مِّن نَّهَارِۭۚ بَلَـٰغࣱۚ فَهَلۡ یُهۡلَكُ إِلَّا ٱلۡقَوۡمُ ٱلۡفَـٰسِقُونَ
