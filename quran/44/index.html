سُورَةُ الدُّخَانِ
Ad-Dukhaan — The Smoke
مكية · 59 آية
الآية 1
بِسۡمِ ٱللَّهِ ٱلرَّحۡمَـٰنِ ٱلرَّحِیمِ حمۤ
الآية 2
وَٱلۡكِتَـٰبِ ٱلۡمُبِینِ
الآية 3
إِنَّاۤ أَنزَلۡنَـٰهُ فِی لَیۡلَةࣲ مُّبَـٰرَكَةٍۚ إِنَّا كُنَّا مُنذِرِینَ
الآية 4
فِیهَا یُفۡرَقُ كُلُّ أَمۡرٍ حَكِیمٍ
الآية 5
أَمۡرࣰا مِّنۡ عِندِنَاۤۚ إِنَّا كُنَّا مُرۡسِلِینَ
الآية 6
رَحۡمَةࣰ مِّن رَّبِّكَۚ إِنَّهُۥ هُوَ ٱلسَّمِیعُ ٱلۡعَلِیمُ
الآية 7
رَبِّ ٱلسَّمَـٰوَ ٰ⁠تِ وَٱلۡأَرۡضِ وَمَا بَیۡنَهُمَاۤۖ إِن كُنتُم مُّوقِنِینَ
الآية 8
لَاۤ إِلَـٰهَ إِلَّا هُوَ یُحۡیِۦ وَیُمِیتُۖ رَبُّكُمۡ وَرَبُّ ءَابَاۤىِٕكُمُ ٱلۡأَوَّلِینَ
الآية 9
بَلۡ هُمۡ فِی شَكࣲّ یَلۡعَبُونَ
الآية 10
فَٱرۡتَقِبۡ یَوۡمَ تَأۡتِی ٱلسَّمَاۤءُ بِدُخَانࣲ مُّبِینࣲ
الآية 11
یَغۡشَى ٱلنَّاسَۖ هَـٰذَا عَذَابٌ أَلِیمࣱ
الآية 12
رَّبَّنَا ٱكۡشِفۡ عَنَّا ٱلۡعَذَابَ إِنَّا مُؤۡمِنُونَ
الآية 13
أَنَّىٰ لَهُمُ ٱلذِّكۡرَىٰ وَقَدۡ جَاۤءَهُمۡ رَسُولࣱ مُّبِینࣱ
الآية 14
ثُمَّ تَوَلَّوۡا۟ عَنۡهُ وَقَالُوا۟ مُعَلَّمࣱ مَّجۡنُونٌ
الآية 15
إِنَّا كَاشِفُوا۟ ٱلۡعَذَابِ قَلِیلًاۚ إِنَّكُمۡ عَاۤىِٕدُونَ
الآية 16
یَوۡمَ نَبۡطِشُ ٱلۡبَطۡشَةَ ٱلۡكُبۡرَىٰۤ إِنَّا مُنتَقِمُونَ
الآية 17
۞ وَلَقَدۡ فَتَنَّا قَبۡلَهُمۡ قَوۡمَ فِرۡعَوۡنَ وَجَاۤءَهُمۡ رَسُولࣱ كَرِیمٌ
الآية 18
أَنۡ أَدُّوۤا۟ إِلَیَّ عِبَادَ ٱللَّهِۖ إِنِّی لَكُمۡ رَسُولٌ أَمِینࣱ
الآية 19
وَأَن لَّا تَعۡلُوا۟ عَلَى ٱللَّهِۖ إِنِّیۤ ءَاتِیكُم بِسُلۡطَـٰنࣲ مُّبِینࣲ
الآية 20
وَإِنِّی عُذۡتُ بِرَبِّی وَرَبِّكُمۡ أَن تَرۡجُمُونِ
الآية 21
وَإِن لَّمۡ تُؤۡمِنُوا۟ لِی فَٱعۡتَزِلُونِ
الآية 22
فَدَعَا رَبَّهُۥۤ أَنَّ هَـٰۤؤُلَاۤءِ قَوۡمࣱ مُّجۡرِمُونَ
الآية 23
فَأَسۡرِ بِعِبَادِی لَیۡلًا إِنَّكُم مُّتَّبَعُونَ
الآية 24
وَٱتۡرُكِ ٱلۡبَحۡرَ رَهۡوًاۖ إِنَّهُمۡ جُندࣱ مُّغۡرَقُونَ
الآية 25
كَمۡ تَرَكُوا۟ مِن جَنَّـٰتࣲ وَعُیُونࣲ
الآية 26
وَزُرُوعࣲ وَمَقَامࣲ كَرِیمࣲ
الآية 27
وَنَعۡمَةࣲ كَانُوا۟ فِیهَا فَـٰكِهِینَ
الآية 28
كَذَ ٰ⁠لِكَۖ وَأَوۡرَثۡنَـٰهَا قَوۡمًا ءَاخَرِینَ
الآية 29
فَمَا بَكَتۡ عَلَیۡهِمُ ٱلسَّمَاۤءُ وَٱلۡأَرۡضُ وَمَا كَانُوا۟ مُنظَرِینَ
الآية 30
وَلَقَدۡ نَجَّیۡنَا بَنِیۤ إِسۡرَ ٰ⁠ۤءِیلَ مِنَ ٱلۡعَذَابِ ٱلۡمُهِینِ
الآية 31
مِن فِرۡعَوۡنَۚ إِنَّهُۥ كَانَ عَالِیࣰا مِّنَ ٱلۡمُسۡرِفِینَ
الآية 32
وَلَقَدِ ٱخۡتَرۡنَـٰهُمۡ عَلَىٰ عِلۡمٍ عَلَى ٱلۡعَـٰلَمِینَ
الآية 33
وَءَاتَیۡنَـٰهُم مِّنَ ٱلۡـَٔایَـٰتِ مَا فِیهِ بَلَـٰۤؤࣱا۟ مُّبِینٌ
الآية 34
إِنَّ هَـٰۤؤُلَاۤءِ لَیَقُولُونَ
الآية 35
إِنۡ هِیَ إِلَّا مَوۡتَتُنَا ٱلۡأُولَىٰ وَمَا نَحۡنُ بِمُنشَرِینَ
الآية 36
فَأۡتُوا۟ بِـَٔابَاۤىِٕنَاۤ إِن كُنتُمۡ صَـٰدِقِینَ
الآية 37
أَهُمۡ خَیۡرٌ أَمۡ قَوۡمُ تُبَّعࣲ وَٱلَّذِینَ مِن قَبۡلِهِمۡ أَهۡلَكۡنَـٰهُمۡۚ إِنَّهُمۡ كَانُوا۟ مُجۡرِمِینَ
الآية 38
وَمَا خَلَقۡنَا ٱلسَّمَـٰوَ ٰ⁠تِ وَٱلۡأَرۡضَ وَمَا بَیۡنَهُمَا لَـٰعِبِینَ
الآية 39
مَا خَلَقۡنَـٰهُمَاۤ إِلَّا بِٱلۡحَقِّ وَلَـٰكِنَّ أَكۡثَرَهُمۡ لَا یَعۡلَمُونَ
الآية 40
إِنَّ یَوۡمَ ٱلۡفَصۡلِ مِیقَـٰتُهُمۡ أَجۡمَعِینَ
الآية 41
یَوۡمَ لَا یُغۡنِی مَوۡلًى عَن مَّوۡلࣰى شَیۡـࣰٔا وَلَا هُمۡ یُنصَرُونَ
الآية 42
إِلَّا مَن رَّحِمَ ٱللَّهُۚ إِنَّهُۥ هُوَ ٱلۡعَزِیزُ ٱلرَّحِیمُ
الآية 43
إِنَّ شَجَرَتَ ٱلزَّقُّومِ
الآية 44
طَعَامُ ٱلۡأَثِیمِ
الآية 45
كَٱلۡمُهۡلِ یَغۡلِی فِی ٱلۡبُطُونِ
الآية 46
كَغَلۡیِ ٱلۡحَمِیمِ
الآية 47
خُذُوهُ فَٱعۡتِلُوهُ إِلَىٰ سَوَاۤءِ ٱلۡجَحِیمِ
الآية 48
ثُمَّ صُبُّوا۟ فَوۡقَ رَأۡسِهِۦ مِنۡ عَذَابِ ٱلۡحَمِیمِ
الآية 49
ذُقۡ إِنَّكَ أَنتَ ٱلۡعَزِیزُ ٱلۡكَرِیمُ
الآية 50
إِنَّ هَـٰذَا مَا كُنتُم بِهِۦ تَمۡتَرُونَ
الآية 51
إِنَّ ٱلۡمُتَّقِینَ فِی مَقَامٍ أَمِینࣲ
الآية 52
فِی جَنَّـٰتࣲ وَعُیُونࣲ
الآية 53
یَلۡبَسُونَ مِن سُندُسࣲ وَإِسۡتَبۡرَقࣲ مُّتَقَـٰبِلِینَ
الآية 54
كَذَ ٰ⁠لِكَ وَزَوَّجۡنَـٰهُم بِحُورٍ عِینࣲ
الآية 55
یَدۡعُونَ فِیهَا بِكُلِّ فَـٰكِهَةٍ ءَامِنِینَ
الآية 56
لَا یَذُوقُونَ فِیهَا ٱلۡمَوۡتَ إِلَّا ٱلۡمَوۡتَةَ ٱلۡأُولَىٰۖ وَوَقَىٰهُمۡ عَذَابَ ٱلۡجَحِیمِ
الآية 57
فَضۡلࣰا مِّن رَّبِّكَۚ ذَ ٰ⁠لِكَ هُوَ ٱلۡفَوۡزُ ٱلۡعَظِیمُ
الآية 58
فَإِنَّمَا یَسَّرۡنَـٰهُ بِلِسَانِكَ لَعَلَّهُمۡ یَتَذَكَّرُونَ
الآية 59
فَٱرۡتَقِبۡ إِنَّهُم مُّرۡتَقِبُونَ
