1
بِسۡمِ ٱللَّهِ ٱلرَّحۡمَـٰنِ ٱلرَّحِیمِ طه
2
مَاۤ أَنزَلۡنَا عَلَیۡكَ ٱلۡقُرۡءَانَ لِتَشۡقَىٰۤ
3
إِلَّا تَذۡكِرَةࣰ لِّمَن یَخۡشَىٰ
4
تَنزِیلࣰا مِّمَّنۡ خَلَقَ ٱلۡأَرۡضَ وَٱلسَّمَـٰوَ ٰ⁠تِ ٱلۡعُلَى
5
ٱلرَّحۡمَـٰنُ عَلَى ٱلۡعَرۡشِ ٱسۡتَوَىٰ
6
لَهُۥ مَا فِی ٱلسَّمَـٰوَ ٰ⁠تِ وَمَا فِی ٱلۡأَرۡضِ وَمَا بَیۡنَهُمَا وَمَا تَحۡتَ ٱلثَّرَىٰ
7
وَإِن تَجۡهَرۡ بِٱلۡقَوۡلِ فَإِنَّهُۥ یَعۡلَمُ ٱلسِّرَّ وَأَخۡفَى
8
ٱللَّهُ لَاۤ إِلَـٰهَ إِلَّا هُوَۖ لَهُ ٱلۡأَسۡمَاۤءُ ٱلۡحُسۡنَىٰ
9
وَهَلۡ أَتَىٰكَ حَدِیثُ مُوسَىٰۤ
10
إِذۡ رَءَا نَارࣰا فَقَالَ لِأَهۡلِهِ ٱمۡكُثُوۤا۟ إِنِّیۤ ءَانَسۡتُ نَارࣰا لَّعَلِّیۤ ءَاتِیكُم مِّنۡهَا بِقَبَسٍ أَوۡ أَجِدُ عَلَى ٱلنَّارِ هُدࣰى
11
فَلَمَّاۤ أَتَىٰهَا نُودِیَ یَـٰمُوسَىٰۤ
12
إِنِّیۤ أَنَا۠ رَبُّكَ فَٱخۡلَعۡ نَعۡلَیۡكَ إِنَّكَ بِٱلۡوَادِ ٱلۡمُقَدَّسِ طُوࣰى
13
وَأَنَا ٱخۡتَرۡتُكَ فَٱسۡتَمِعۡ لِمَا یُوحَىٰۤ
14
إِنَّنِیۤ أَنَا ٱللَّهُ لَاۤ إِلَـٰهَ إِلَّاۤ أَنَا۠ فَٱعۡبُدۡنِی وَأَقِمِ ٱلصَّلَوٰةَ لِذِكۡرِیۤ
15
إِنَّ ٱلسَّاعَةَ ءَاتِیَةٌ أَكَادُ أُخۡفِیهَا لِتُجۡزَىٰ كُلُّ نَفۡسِۭ بِمَا تَسۡعَىٰ
16
فَلَا یَصُدَّنَّكَ عَنۡهَا مَن لَّا یُؤۡمِنُ بِهَا وَٱتَّبَعَ هَوَىٰهُ فَتَرۡدَىٰ
17
وَمَا تِلۡكَ بِیَمِینِكَ یَـٰمُوسَىٰ
18
قَالَ هِیَ عَصَایَ أَتَوَكَّؤُا۟ عَلَیۡهَا وَأَهُشُّ بِهَا عَلَىٰ غَنَمِی وَلِیَ فِیهَا مَـَٔارِبُ أُخۡرَىٰ
19
قَالَ أَلۡقِهَا یَـٰمُوسَىٰ
20
فَأَلۡقَىٰهَا فَإِذَا هِیَ حَیَّةࣱ تَسۡعَىٰ
21
قَالَ خُذۡهَا وَلَا تَخَفۡۖ سَنُعِیدُهَا سِیرَتَهَا ٱلۡأُولَىٰ
22
وَٱضۡمُمۡ یَدَكَ إِلَىٰ جَنَاحِكَ تَخۡرُجۡ بَیۡضَاۤءَ مِنۡ غَیۡرِ سُوۤءٍ ءَایَةً أُخۡرَىٰ
23
لِنُرِیَكَ مِنۡ ءَایَـٰتِنَا ٱلۡكُبۡرَى
24
ٱذۡهَبۡ إِلَىٰ فِرۡعَوۡنَ إِنَّهُۥ طَغَىٰ
25
قَالَ رَبِّ ٱشۡرَحۡ لِی صَدۡرِی
26
وَیَسِّرۡ لِیۤ أَمۡرِی
27
وَٱحۡلُلۡ عُقۡدَةࣰ مِّن لِّسَانِی
28
یَفۡقَهُوا۟ قَوۡلِی
29
وَٱجۡعَل لِّی وَزِیرࣰا مِّنۡ أَهۡلِی
30
هَـٰرُونَ أَخِی
31
ٱشۡدُدۡ بِهِۦۤ أَزۡرِی
32
وَأَشۡرِكۡهُ فِیۤ أَمۡرِی
33
كَیۡ نُسَبِّحَكَ كَثِیرࣰا
34
وَنَذۡكُرَكَ كَثِیرًا
35
إِنَّكَ كُنتَ بِنَا بَصِیرࣰا
36
قَالَ قَدۡ أُوتِیتَ سُؤۡلَكَ یَـٰمُوسَىٰ
37
وَلَقَدۡ مَنَنَّا عَلَیۡكَ مَرَّةً أُخۡرَىٰۤ
38
إِذۡ أَوۡحَیۡنَاۤ إِلَىٰۤ أُمِّكَ مَا یُوحَىٰۤ
39
أَنِ ٱقۡذِفِیهِ فِی ٱلتَّابُوتِ فَٱقۡذِفِیهِ فِی ٱلۡیَمِّ فَلۡیُلۡقِهِ ٱلۡیَمُّ بِٱلسَّاحِلِ یَأۡخُذۡهُ عَدُوࣱّ لِّی وَعَدُوࣱّ لَّهُۥۚ وَأَلۡقَیۡتُ عَلَیۡكَ مَحَبَّةࣰ مِّنِّی وَلِتُصۡنَعَ عَلَىٰ عَیۡنِیۤ
40
إِذۡ تَمۡشِیۤ أُخۡتُكَ فَتَقُولُ هَلۡ أَدُلُّكُمۡ عَلَىٰ مَن یَكۡفُلُهُۥۖ فَرَجَعۡنَـٰكَ إِلَىٰۤ أُمِّكَ كَیۡ تَقَرَّ عَیۡنُهَا وَلَا تَحۡزَنَۚ وَقَتَلۡتَ نَفۡسࣰا فَنَجَّیۡنَـٰكَ مِنَ ٱلۡغَمِّ وَفَتَنَّـٰكَ فُتُونࣰاۚ فَلَبِثۡتَ سِنِینَ فِیۤ أَهۡلِ مَدۡیَنَ ثُمَّ جِئۡتَ عَلَىٰ قَدَرࣲ یَـٰمُوسَىٰ
41
وَٱصۡطَنَعۡتُكَ لِنَفۡسِی
42
ٱذۡهَبۡ أَنتَ وَأَخُوكَ بِـَٔایَـٰتِی وَلَا تَنِیَا فِی ذِكۡرِی
43
ٱذۡهَبَاۤ إِلَىٰ فِرۡعَوۡنَ إِنَّهُۥ طَغَىٰ
44
فَقُولَا لَهُۥ قَوۡلࣰا لَّیِّنࣰا لَّعَلَّهُۥ یَتَذَكَّرُ أَوۡ یَخۡشَىٰ
45
قَالَا رَبَّنَاۤ إِنَّنَا نَخَافُ أَن یَفۡرُطَ عَلَیۡنَاۤ أَوۡ أَن یَطۡغَىٰ
46
قَالَ لَا تَخَافَاۤۖ إِنَّنِی مَعَكُمَاۤ أَسۡمَعُ وَأَرَىٰ
47
فَأۡتِیَاهُ فَقُولَاۤ إِنَّا رَسُولَا رَبِّكَ فَأَرۡسِلۡ مَعَنَا بَنِیۤ إِسۡرَ ٰ⁠ۤءِیلَ وَلَا تُعَذِّبۡهُمۡۖ قَدۡ جِئۡنَـٰكَ بِـَٔایَةࣲ مِّن رَّبِّكَۖ وَٱلسَّلَـٰمُ عَلَىٰ مَنِ ٱتَّبَعَ ٱلۡهُدَىٰۤ
48
إِنَّا قَدۡ أُوحِیَ إِلَیۡنَاۤ أَنَّ ٱلۡعَذَابَ عَلَىٰ مَن كَذَّبَ وَتَوَلَّىٰ
49
قَالَ فَمَن رَّبُّكُمَا یَـٰمُوسَىٰ
50
قَالَ رَبُّنَا ٱلَّذِیۤ أَعۡطَىٰ كُلَّ شَیۡءٍ خَلۡقَهُۥ ثُمَّ هَدَىٰ
51
قَالَ فَمَا بَالُ ٱلۡقُرُونِ ٱلۡأُولَىٰ
52
قَالَ عِلۡمُهَا عِندَ رَبِّی فِی كِتَـٰبࣲۖ لَّا یَضِلُّ رَبِّی وَلَا یَنسَى
53
ٱلَّذِی جَعَلَ لَكُمُ ٱلۡأَرۡضَ مَهۡدࣰا وَسَلَكَ لَكُمۡ فِیهَا سُبُلࣰا وَأَنزَلَ مِنَ ٱلسَّمَاۤءِ مَاۤءࣰ فَأَخۡرَجۡنَا بِهِۦۤ أَزۡوَ ٰ⁠جࣰا مِّن نَّبَاتࣲ شَتَّىٰ
54
كُلُوا۟ وَٱرۡعَوۡا۟ أَنۡعَـٰمَكُمۡۚ إِنَّ فِی ذَ ٰ⁠لِكَ لَـَٔایَـٰتࣲ لِّأُو۟لِی ٱلنُّهَىٰ
55
۞ مِنۡهَا خَلَقۡنَـٰكُمۡ وَفِیهَا نُعِیدُكُمۡ وَمِنۡهَا نُخۡرِجُكُمۡ تَارَةً أُخۡرَىٰ
56
وَلَقَدۡ أَرَیۡنَـٰهُ ءَایَـٰتِنَا كُلَّهَا فَكَذَّبَ وَأَبَىٰ
57
قَالَ أَجِئۡتَنَا لِتُخۡرِجَنَا مِنۡ أَرۡضِنَا بِسِحۡرِكَ یَـٰمُوسَىٰ
58
فَلَنَأۡتِیَنَّكَ بِسِحۡرࣲ مِّثۡلِهِۦ فَٱجۡعَلۡ بَیۡنَنَا وَبَیۡنَكَ مَوۡعِدࣰا لَّا نُخۡلِفُهُۥ نَحۡنُ وَلَاۤ أَنتَ مَكَانࣰا سُوࣰى
59
قَالَ مَوۡعِدُكُمۡ یَوۡمُ ٱلزِّینَةِ وَأَن یُحۡشَرَ ٱلنَّاسُ ضُحࣰى
60
فَتَوَلَّىٰ فِرۡعَوۡنُ فَجَمَعَ كَیۡدَهُۥ ثُمَّ أَتَىٰ
61
قَالَ لَهُم مُّوسَىٰ وَیۡلَكُمۡ لَا تَفۡتَرُوا۟ عَلَى ٱللَّهِ كَذِبࣰا فَیُسۡحِتَكُم بِعَذَابࣲۖ وَقَدۡ خَابَ مَنِ ٱفۡتَرَىٰ
62
فَتَنَـٰزَعُوۤا۟ أَمۡرَهُم بَیۡنَهُمۡ وَأَسَرُّوا۟ ٱلنَّجۡوَىٰ
63
قَالُوۤا۟ إِنۡ هَـٰذَ ٰ⁠نِ لَسَـٰحِرَ ٰ⁠نِ یُرِیدَانِ أَن یُخۡرِجَاكُم مِّنۡ أَرۡضِكُم بِسِحۡرِهِمَا وَیَذۡهَبَا بِطَرِیقَتِكُمُ ٱلۡمُثۡلَىٰ
64
فَأَجۡمِعُوا۟ كَیۡدَكُمۡ ثُمَّ ٱئۡتُوا۟ صَفࣰّاۚ وَقَدۡ أَفۡلَحَ ٱلۡیَوۡمَ مَنِ ٱسۡتَعۡلَىٰ
65
قَالُوا۟ یَـٰمُوسَىٰۤ إِمَّاۤ أَن تُلۡقِیَ وَإِمَّاۤ أَن نَّكُونَ أَوَّلَ مَنۡ أَلۡقَىٰ
66
قَالَ بَلۡ أَلۡقُوا۟ۖ فَإِذَا حِبَالُهُمۡ وَعِصِیُّهُمۡ یُخَیَّلُ إِلَیۡهِ مِن سِحۡرِهِمۡ أَنَّهَا تَسۡعَىٰ
67
فَأَوۡجَسَ فِی نَفۡسِهِۦ خِیفَةࣰ مُّوسَىٰ
68
قُلۡنَا لَا تَخَفۡ إِنَّكَ أَنتَ ٱلۡأَعۡلَىٰ
69
وَأَلۡقِ مَا فِی یَمِینِكَ تَلۡقَفۡ مَا صَنَعُوۤا۟ۖ إِنَّمَا صَنَعُوا۟ كَیۡدُ سَـٰحِرࣲۖ وَلَا یُفۡلِحُ ٱلسَّاحِرُ حَیۡثُ أَتَىٰ
70
فَأُلۡقِیَ ٱلسَّحَرَةُ سُجَّدࣰا قَالُوۤا۟ ءَامَنَّا بِرَبِّ هَـٰرُونَ وَمُوسَىٰ
71
قَالَ ءَامَنتُمۡ لَهُۥ قَبۡلَ أَنۡ ءَاذَنَ لَكُمۡۖ إِنَّهُۥ لَكَبِیرُكُمُ ٱلَّذِی عَلَّمَكُمُ ٱلسِّحۡرَۖ فَلَأُقَطِّعَنَّ أَیۡدِیَكُمۡ وَأَرۡجُلَكُم مِّنۡ خِلَـٰفࣲ وَلَأُصَلِّبَنَّكُمۡ فِی جُذُوعِ ٱلنَّخۡلِ وَلَتَعۡلَمُنَّ أَیُّنَاۤ أَشَدُّ عَذَابࣰا وَأَبۡقَىٰ
72
قَالُوا۟ لَن نُّؤۡثِرَكَ عَلَىٰ مَا جَاۤءَنَا مِنَ ٱلۡبَیِّنَـٰتِ وَٱلَّذِی فَطَرَنَاۖ فَٱقۡضِ مَاۤ أَنتَ قَاضٍۖ إِنَّمَا تَقۡضِی هَـٰذِهِ ٱلۡحَیَوٰةَ ٱلدُّنۡیَاۤ
73
إِنَّاۤ ءَامَنَّا بِرَبِّنَا لِیَغۡفِرَ لَنَا خَطَـٰیَـٰنَا وَمَاۤ أَكۡرَهۡتَنَا عَلَیۡهِ مِنَ ٱلسِّحۡرِۗ وَٱللَّهُ خَیۡرࣱ وَأَبۡقَىٰۤ
74
إِنَّهُۥ مَن یَأۡتِ رَبَّهُۥ مُجۡرِمࣰا فَإِنَّ لَهُۥ جَهَنَّمَ لَا یَمُوتُ فِیهَا وَلَا یَحۡیَىٰ
75
وَمَن یَأۡتِهِۦ مُؤۡمِنࣰا قَدۡ عَمِلَ ٱلصَّـٰلِحَـٰتِ فَأُو۟لَـٰۤىِٕكَ لَهُمُ ٱلدَّرَجَـٰتُ ٱلۡعُلَىٰ
76
جَنَّـٰتُ عَدۡنࣲ تَجۡرِی مِن تَحۡتِهَا ٱلۡأَنۡهَـٰرُ خَـٰلِدِینَ فِیهَاۚ وَذَ ٰ⁠لِكَ جَزَاۤءُ مَن تَزَكَّىٰ
77
وَلَقَدۡ أَوۡحَیۡنَاۤ إِلَىٰ مُوسَىٰۤ أَنۡ أَسۡرِ بِعِبَادِی فَٱضۡرِبۡ لَهُمۡ طَرِیقࣰا فِی ٱلۡبَحۡرِ یَبَسࣰا لَّا تَخَـٰفُ دَرَكࣰا وَلَا تَخۡشَىٰ
78
فَأَتۡبَعَهُمۡ فِرۡعَوۡنُ بِجُنُودِهِۦ فَغَشِیَهُم مِّنَ ٱلۡیَمِّ مَا غَشِیَهُمۡ
79
وَأَضَلَّ فِرۡعَوۡنُ قَوۡمَهُۥ وَمَا هَدَىٰ
80
یَـٰبَنِیۤ إِسۡرَ ٰ⁠ۤءِیلَ قَدۡ أَنجَیۡنَـٰكُم مِّنۡ عَدُوِّكُمۡ وَوَ ٰ⁠عَدۡنَـٰكُمۡ جَانِبَ ٱلطُّورِ ٱلۡأَیۡمَنَ وَنَزَّلۡنَا عَلَیۡكُمُ ٱلۡمَنَّ وَٱلسَّلۡوَىٰ
81
كُلُوا۟ مِن طَیِّبَـٰتِ مَا رَزَقۡنَـٰكُمۡ وَلَا تَطۡغَوۡا۟ فِیهِ فَیَحِلَّ عَلَیۡكُمۡ غَضَبِیۖ وَمَن یَحۡلِلۡ عَلَیۡهِ غَضَبِی فَقَدۡ هَوَىٰ
82
وَإِنِّی لَغَفَّارࣱ لِّمَن تَابَ وَءَامَنَ وَعَمِلَ صَـٰلِحࣰا ثُمَّ ٱهۡتَدَىٰ
83
۞ وَمَاۤ أَعۡجَلَكَ عَن قَوۡمِكَ یَـٰمُوسَىٰ
84
قَالَ هُمۡ أُو۟لَاۤءِ عَلَىٰۤ أَثَرِی وَعَجِلۡتُ إِلَیۡكَ رَبِّ لِتَرۡضَىٰ
85
قَالَ فَإِنَّا قَدۡ فَتَنَّا قَوۡمَكَ مِنۢ بَعۡدِكَ وَأَضَلَّهُمُ ٱلسَّامِرِیُّ
86
فَرَجَعَ مُوسَىٰۤ إِلَىٰ قَوۡمِهِۦ غَضۡبَـٰنَ أَسِفࣰاۚ قَالَ یَـٰقَوۡمِ أَلَمۡ یَعِدۡكُمۡ رَبُّكُمۡ وَعۡدًا حَسَنًاۚ أَفَطَالَ عَلَیۡكُمُ ٱلۡعَهۡدُ أَمۡ أَرَدتُّمۡ أَن یَحِلَّ عَلَیۡكُمۡ غَضَبࣱ مِّن رَّبِّكُمۡ فَأَخۡلَفۡتُم مَّوۡعِدِی
87
قَالُوا۟ مَاۤ أَخۡلَفۡنَا مَوۡعِدَكَ بِمَلۡكِنَا وَلَـٰكِنَّا حُمِّلۡنَاۤ أَوۡزَارࣰا مِّن زِینَةِ ٱلۡقَوۡمِ فَقَذَفۡنَـٰهَا فَكَذَ ٰ⁠لِكَ أَلۡقَى ٱلسَّامِرِیُّ
88
فَأَخۡرَجَ لَهُمۡ عِجۡلࣰا جَسَدࣰا لَّهُۥ خُوَارࣱ فَقَالُوا۟ هَـٰذَاۤ إِلَـٰهُكُمۡ وَإِلَـٰهُ مُوسَىٰ فَنَسِیَ
89
أَفَلَا یَرَوۡنَ أَلَّا یَرۡجِعُ إِلَیۡهِمۡ قَوۡلࣰا وَلَا یَمۡلِكُ لَهُمۡ ضَرࣰّا وَلَا نَفۡعࣰا
90
وَلَقَدۡ قَالَ لَهُمۡ هَـٰرُونُ مِن قَبۡلُ یَـٰقَوۡمِ إِنَّمَا فُتِنتُم بِهِۦۖ وَإِنَّ رَبَّكُمُ ٱلرَّحۡمَـٰنُ فَٱتَّبِعُونِی وَأَطِیعُوۤا۟ أَمۡرِی
91
قَالُوا۟ لَن نَّبۡرَحَ عَلَیۡهِ عَـٰكِفِینَ حَتَّىٰ یَرۡجِعَ إِلَیۡنَا مُوسَىٰ
92
قَالَ یَـٰهَـٰرُونُ مَا مَنَعَكَ إِذۡ رَأَیۡتَهُمۡ ضَلُّوۤا۟
93
أَلَّا تَتَّبِعَنِۖ أَفَعَصَیۡتَ أَمۡرِی
94
قَالَ یَبۡنَؤُمَّ لَا تَأۡخُذۡ بِلِحۡیَتِی وَلَا بِرَأۡسِیۤۖ إِنِّی خَشِیتُ أَن تَقُولَ فَرَّقۡتَ بَیۡنَ بَنِیۤ إِسۡرَ ٰ⁠ۤءِیلَ وَلَمۡ تَرۡقُبۡ قَوۡلِی
95
قَالَ فَمَا خَطۡبُكَ یَـٰسَـٰمِرِیُّ
96
قَالَ بَصُرۡتُ بِمَا لَمۡ یَبۡصُرُوا۟ بِهِۦ فَقَبَضۡتُ قَبۡضَةࣰ مِّنۡ أَثَرِ ٱلرَّسُولِ فَنَبَذۡتُهَا وَكَذَ ٰ⁠لِكَ سَوَّلَتۡ لِی نَفۡسِی
97
قَالَ فَٱذۡهَبۡ فَإِنَّ لَكَ فِی ٱلۡحَیَوٰةِ أَن تَقُولَ لَا مِسَاسَۖ وَإِنَّ لَكَ مَوۡعِدࣰا لَّن تُخۡلَفَهُۥۖ وَٱنظُرۡ إِلَىٰۤ إِلَـٰهِكَ ٱلَّذِی ظَلۡتَ عَلَیۡهِ عَاكِفࣰاۖ لَّنُحَرِّقَنَّهُۥ ثُمَّ لَنَنسِفَنَّهُۥ فِی ٱلۡیَمِّ نَسۡفًا
98
إِنَّمَاۤ إِلَـٰهُكُمُ ٱللَّهُ ٱلَّذِی لَاۤ إِلَـٰهَ إِلَّا هُوَۚ وَسِعَ كُلَّ شَیۡءٍ عِلۡمࣰا
99
كَذَ ٰ⁠لِكَ نَقُصُّ عَلَیۡكَ مِنۡ أَنۢبَاۤءِ مَا قَدۡ سَبَقَۚ وَقَدۡ ءَاتَیۡنَـٰكَ مِن لَّدُنَّا ذِكۡرࣰا
100
مَّنۡ أَعۡرَضَ عَنۡهُ فَإِنَّهُۥ یَحۡمِلُ یَوۡمَ ٱلۡقِیَـٰمَةِ وِزۡرًا
101
خَـٰلِدِینَ فِیهِۖ وَسَاۤءَ لَهُمۡ یَوۡمَ ٱلۡقِیَـٰمَةِ حِمۡلࣰا
102
یَوۡمَ یُنفَخُ فِی ٱلصُّورِۚ وَنَحۡشُرُ ٱلۡمُجۡرِمِینَ یَوۡمَىِٕذࣲ زُرۡقࣰا
103
یَتَخَـٰفَتُونَ بَیۡنَهُمۡ إِن لَّبِثۡتُمۡ إِلَّا عَشۡرࣰا
104
نَّحۡنُ أَعۡلَمُ بِمَا یَقُولُونَ إِذۡ یَقُولُ أَمۡثَلُهُمۡ طَرِیقَةً إِن لَّبِثۡتُمۡ إِلَّا یَوۡمࣰا
105
وَیَسۡـَٔلُونَكَ عَنِ ٱلۡجِبَالِ فَقُلۡ یَنسِفُهَا رَبِّی نَسۡفࣰا
106
فَیَذَرُهَا قَاعࣰا صَفۡصَفࣰا
107
لَّا تَرَىٰ فِیهَا عِوَجࣰا وَلَاۤ أَمۡتࣰا
108
یَوۡمَىِٕذࣲ یَتَّبِعُونَ ٱلدَّاعِیَ لَا عِوَجَ لَهُۥۖ وَخَشَعَتِ ٱلۡأَصۡوَاتُ لِلرَّحۡمَـٰنِ فَلَا تَسۡمَعُ إِلَّا هَمۡسࣰا
109
یَوۡمَىِٕذࣲ لَّا تَنفَعُ ٱلشَّفَـٰعَةُ إِلَّا مَنۡ أَذِنَ لَهُ ٱلرَّحۡمَـٰنُ وَرَضِیَ لَهُۥ قَوۡلࣰا
110
یَعۡلَمُ مَا بَیۡنَ أَیۡدِیهِمۡ وَمَا خَلۡفَهُمۡ وَلَا یُحِیطُونَ بِهِۦ عِلۡمࣰا
111
۞ وَعَنَتِ ٱلۡوُجُوهُ لِلۡحَیِّ ٱلۡقَیُّومِۖ وَقَدۡ خَابَ مَنۡ حَمَلَ ظُلۡمࣰا
112
وَمَن یَعۡمَلۡ مِنَ ٱلصَّـٰلِحَـٰتِ وَهُوَ مُؤۡمِنࣱ فَلَا یَخَافُ ظُلۡمࣰا وَلَا هَضۡمࣰا
113
وَكَذَ ٰ⁠لِكَ أَنزَلۡنَـٰهُ قُرۡءَانًا عَرَبِیࣰّا وَصَرَّفۡنَا فِیهِ مِنَ ٱلۡوَعِیدِ لَعَلَّهُمۡ یَتَّقُونَ أَوۡ یُحۡدِثُ لَهُمۡ ذِكۡرࣰا
114
فَتَعَـٰلَى ٱللَّهُ ٱلۡمَلِكُ ٱلۡحَقُّۗ وَلَا تَعۡجَلۡ بِٱلۡقُرۡءَانِ مِن قَبۡلِ أَن یُقۡضَىٰۤ إِلَیۡكَ وَحۡیُهُۥۖ وَقُل رَّبِّ زِدۡنِی عِلۡمࣰا
115
وَلَقَدۡ عَهِدۡنَاۤ إِلَىٰۤ ءَادَمَ مِن قَبۡلُ فَنَسِیَ وَلَمۡ نَجِدۡ لَهُۥ عَزۡمࣰا
116
وَإِذۡ قُلۡنَا لِلۡمَلَـٰۤىِٕكَةِ ٱسۡجُدُوا۟ لِـَٔادَمَ فَسَجَدُوۤا۟ إِلَّاۤ إِبۡلِیسَ أَبَىٰ
117
فَقُلۡنَا یَـٰۤـَٔادَمُ إِنَّ هَـٰذَا عَدُوࣱّ لَّكَ وَلِزَوۡجِكَ فَلَا یُخۡرِجَنَّكُمَا مِنَ ٱلۡجَنَّةِ فَتَشۡقَىٰۤ
118
إِنَّ لَكَ أَلَّا تَجُوعَ فِیهَا وَلَا تَعۡرَىٰ
119
وَأَنَّكَ لَا تَظۡمَؤُا۟ فِیهَا وَلَا تَضۡحَىٰ
120
فَوَسۡوَسَ إِلَیۡهِ ٱلشَّیۡطَـٰنُ قَالَ یَـٰۤـَٔادَمُ هَلۡ أَدُلُّكَ عَلَىٰ شَجَرَةِ ٱلۡخُلۡدِ وَمُلۡكࣲ لَّا یَبۡلَىٰ
121
فَأَكَلَا مِنۡهَا فَبَدَتۡ لَهُمَا سَوۡءَ ٰ⁠ تُهُمَا وَطَفِقَا یَخۡصِفَانِ عَلَیۡهِمَا مِن وَرَقِ ٱلۡجَنَّةِۚ وَعَصَىٰۤ ءَادَمُ رَبَّهُۥ فَغَوَىٰ
122
ثُمَّ ٱجۡتَبَـٰهُ رَبُّهُۥ فَتَابَ عَلَیۡهِ وَهَدَىٰ
123
قَالَ ٱهۡبِطَا مِنۡهَا جَمِیعَۢاۖ بَعۡضُكُمۡ لِبَعۡضٍ عَدُوࣱّۖ فَإِمَّا یَأۡتِیَنَّكُم مِّنِّی هُدࣰى فَمَنِ ٱتَّبَعَ هُدَایَ فَلَا یَضِلُّ وَلَا یَشۡقَىٰ
124
وَمَنۡ أَعۡرَضَ عَن ذِكۡرِی فَإِنَّ لَهُۥ مَعِیشَةࣰ ضَنكࣰا وَنَحۡشُرُهُۥ یَوۡمَ ٱلۡقِیَـٰمَةِ أَعۡمَىٰ
125
قَالَ رَبِّ لِمَ حَشَرۡتَنِیۤ أَعۡمَىٰ وَقَدۡ كُنتُ بَصِیرࣰا
126
قَالَ كَذَ ٰ⁠لِكَ أَتَتۡكَ ءَایَـٰتُنَا فَنَسِیتَهَاۖ وَكَذَ ٰ⁠لِكَ ٱلۡیَوۡمَ تُنسَىٰ
127
وَكَذَ ٰ⁠لِكَ نَجۡزِی مَنۡ أَسۡرَفَ وَلَمۡ یُؤۡمِنۢ بِـَٔایَـٰتِ رَبِّهِۦۚ وَلَعَذَابُ ٱلۡـَٔاخِرَةِ أَشَدُّ وَأَبۡقَىٰۤ
128
أَفَلَمۡ یَهۡدِ لَهُمۡ كَمۡ أَهۡلَكۡنَا قَبۡلَهُم مِّنَ ٱلۡقُرُونِ یَمۡشُونَ فِی مَسَـٰكِنِهِمۡۚ إِنَّ فِی ذَ ٰ⁠لِكَ لَـَٔایَـٰتࣲ لِّأُو۟لِی ٱلنُّهَىٰ
129
وَلَوۡلَا كَلِمَةࣱ سَبَقَتۡ مِن رَّبِّكَ لَكَانَ لِزَامࣰا وَأَجَلࣱ مُّسَمࣰّى
130
فَٱصۡبِرۡ عَلَىٰ مَا یَقُولُونَ وَسَبِّحۡ بِحَمۡدِ رَبِّكَ قَبۡلَ طُلُوعِ ٱلشَّمۡسِ وَقَبۡلَ غُرُوبِهَاۖ وَمِنۡ ءَانَاۤىِٕ ٱلَّیۡلِ فَسَبِّحۡ وَأَطۡرَافَ ٱلنَّهَارِ لَعَلَّكَ تَرۡضَىٰ
131
وَلَا تَمُدَّنَّ عَیۡنَیۡكَ إِلَىٰ مَا مَتَّعۡنَا بِهِۦۤ أَزۡوَ ٰ⁠جࣰا مِّنۡهُمۡ زَهۡرَةَ ٱلۡحَیَوٰةِ ٱلدُّنۡیَا لِنَفۡتِنَهُمۡ فِیهِۚ وَرِزۡقُ رَبِّكَ خَیۡرࣱ وَأَبۡقَىٰ
132
وَأۡمُرۡ أَهۡلَكَ بِٱلصَّلَوٰةِ وَٱصۡطَبِرۡ عَلَیۡهَاۖ لَا نَسۡـَٔلُكَ رِزۡقࣰاۖ نَّحۡنُ نَرۡزُقُكَۗ وَٱلۡعَـٰقِبَةُ لِلتَّقۡوَىٰ
133
وَقَالُوا۟ لَوۡلَا یَأۡتِینَا بِـَٔایَةࣲ مِّن رَّبِّهِۦۤۚ أَوَلَمۡ تَأۡتِهِم بَیِّنَةُ مَا فِی ٱلصُّحُفِ ٱلۡأُولَىٰ
134
وَلَوۡ أَنَّاۤ أَهۡلَكۡنَـٰهُم بِعَذَابࣲ مِّن قَبۡلِهِۦ لَقَالُوا۟ رَبَّنَا لَوۡلَاۤ أَرۡسَلۡتَ إِلَیۡنَا رَسُولࣰا فَنَتَّبِعَ ءَایَـٰتِكَ مِن قَبۡلِ أَن نَّذِلَّ وَنَخۡزَىٰ
135
قُلۡ كُلࣱّ مُّتَرَبِّصࣱ فَتَرَبَّصُوا۟ۖ فَسَتَعۡلَمُونَ مَنۡ أَصۡحَـٰبُ ٱلصِّرَ ٰ⁠طِ ٱلسَّوِیِّ وَمَنِ ٱهۡتَدَىٰ
