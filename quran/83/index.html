سُورَةُ المُطَفِّفِينَ
Al-Mutaffifin — Defrauding
مكية · 36 آية
الآية 1
بِسۡمِ ٱللَّهِ ٱلرَّحۡمَـٰنِ ٱلرَّحِیمِ وَیۡلࣱ لِّلۡمُطَفِّفِینَ
الآية 2
ٱلَّذِینَ إِذَا ٱكۡتَالُوا۟ عَلَى ٱلنَّاسِ یَسۡتَوۡفُونَ
الآية 3
وَإِذَا كَالُوهُمۡ أَو وَّزَنُوهُمۡ یُخۡسِرُونَ
الآية 4
أَلَا یَظُنُّ أُو۟لَـٰۤىِٕكَ أَنَّهُم مَّبۡعُوثُونَ
الآية 5
لِیَوۡمٍ عَظِیمࣲ
الآية 6
یَوۡمَ یَقُومُ ٱلنَّاسُ لِرَبِّ ٱلۡعَـٰلَمِینَ
الآية 7
كَلَّاۤ إِنَّ كِتَـٰبَ ٱلۡفُجَّارِ لَفِی سِجِّینࣲ
الآية 8
وَمَاۤ أَدۡرَىٰكَ مَا سِجِّینࣱ
الآية 9
كِتَـٰبࣱ مَّرۡقُومࣱ
الآية 10
وَیۡلࣱ یَوۡمَىِٕذࣲ لِّلۡمُكَذِّبِینَ
الآية 11
ٱلَّذِینَ یُكَذِّبُونَ بِیَوۡمِ ٱلدِّینِ
الآية 12
وَمَا یُكَذِّبُ بِهِۦۤ إِلَّا كُلُّ مُعۡتَدٍ أَثِیمٍ
الآية 13
إِذَا تُتۡلَىٰ عَلَیۡهِ ءَایَـٰتُنَا قَالَ أَسَـٰطِیرُ ٱلۡأَوَّلِینَ
الآية 14
كَلَّاۖ بَلۡۜ رَانَ عَلَىٰ قُلُوبِهِم مَّا كَانُوا۟ یَكۡسِبُونَ
الآية 15
كَلَّاۤ إِنَّهُمۡ عَن رَّبِّهِمۡ یَوۡمَىِٕذࣲ لَّمَحۡجُوبُونَ
الآية 16
ثُمَّ إِنَّهُمۡ لَصَالُوا۟ ٱلۡجَحِیمِ
الآية 17
ثُمَّ یُقَالُ هَـٰذَا ٱلَّذِی كُنتُم بِهِۦ تُكَذِّبُونَ
الآية 18
كَلَّاۤ إِنَّ كِتَـٰبَ ٱلۡأَبۡرَارِ لَفِی عِلِّیِّینَ
الآية 19
وَمَاۤ أَدۡرَىٰكَ مَا عِلِّیُّونَ
الآية 20
كِتَـٰبࣱ مَّرۡقُومࣱ
الآية 21
یَشۡهَدُهُ ٱلۡمُقَرَّبُونَ
الآية 22
إِنَّ ٱلۡأَبۡرَارَ لَفِی نَعِیمٍ
الآية 23
عَلَى ٱلۡأَرَاۤىِٕكِ یَنظُرُونَ
الآية 24
تَعۡرِفُ فِی وُجُوهِهِمۡ نَضۡرَةَ ٱلنَّعِیمِ
الآية 25
یُسۡقَوۡنَ مِن رَّحِیقࣲ مَّخۡتُومٍ
الآية 26
خِتَـٰمُهُۥ مِسۡكࣱۚ وَفِی ذَ ٰ⁠لِكَ فَلۡیَتَنَافَسِ ٱلۡمُتَنَـٰفِسُونَ
الآية 27
وَمِزَاجُهُۥ مِن تَسۡنِیمٍ
الآية 28
عَیۡنࣰا یَشۡرَبُ بِهَا ٱلۡمُقَرَّبُونَ
الآية 29
إِنَّ ٱلَّذِینَ أَجۡرَمُوا۟ كَانُوا۟ مِنَ ٱلَّذِینَ ءَامَنُوا۟ یَضۡحَكُونَ
الآية 30
وَإِذَا مَرُّوا۟ بِهِمۡ یَتَغَامَزُونَ
الآية 31
وَإِذَا ٱنقَلَبُوۤا۟ إِلَىٰۤ أَهۡلِهِمُ ٱنقَلَبُوا۟ فَكِهِینَ
الآية 32
وَإِذَا رَأَوۡهُمۡ قَالُوۤا۟ إِنَّ هَـٰۤؤُلَاۤءِ لَضَاۤلُّونَ
الآية 33
وَمَاۤ أُرۡسِلُوا۟ عَلَیۡهِمۡ حَـٰفِظِینَ
الآية 34
فَٱلۡیَوۡمَ ٱلَّذِینَ ءَامَنُوا۟ مِنَ ٱلۡكُفَّارِ یَضۡحَكُونَ
الآية 35
عَلَى ٱلۡأَرَاۤىِٕكِ یَنظُرُونَ
الآية 36
هَلۡ ثُوِّبَ ٱلۡكُفَّارُ مَا كَانُوا۟ یَفۡعَلُونَ
