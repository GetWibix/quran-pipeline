1
بِسۡمِ ٱللَّهِ ٱلرَّحۡمَـٰنِ ٱلرَّحِیمِ طسۤمۤ
2
تِلۡكَ ءَایَـٰتُ ٱلۡكِتَـٰبِ ٱلۡمُبِینِ
3
لَعَلَّكَ بَـٰخِعࣱ نَّفۡسَكَ أَلَّا یَكُونُوا۟ مُؤۡمِنِینَ
4
إِن نَّشَأۡ نُنَزِّلۡ عَلَیۡهِم مِّنَ ٱلسَّمَاۤءِ ءَایَةࣰ فَظَلَّتۡ أَعۡنَـٰقُهُمۡ لَهَا خَـٰضِعِینَ
5
وَمَا یَأۡتِیهِم مِّن ذِكۡرࣲ مِّنَ ٱلرَّحۡمَـٰنِ مُحۡدَثٍ إِلَّا كَانُوا۟ عَنۡهُ مُعۡرِضِینَ
6
فَقَدۡ كَذَّبُوا۟ فَسَیَأۡتِیهِمۡ أَنۢبَـٰۤؤُا۟ مَا كَانُوا۟ بِهِۦ یَسۡتَهۡزِءُونَ
7
أَوَلَمۡ یَرَوۡا۟ إِلَى ٱلۡأَرۡضِ كَمۡ أَنۢبَتۡنَا فِیهَا مِن كُلِّ زَوۡجࣲ كَرِیمٍ
8
إِنَّ فِی ذَ ٰ⁠لِكَ لَـَٔایَةࣰۖ وَمَا كَانَ أَكۡثَرُهُم مُّؤۡمِنِینَ
9
وَإِنَّ رَبَّكَ لَهُوَ ٱلۡعَزِیزُ ٱلرَّحِیمُ
10
وَإِذۡ نَادَىٰ رَبُّكَ مُوسَىٰۤ أَنِ ٱئۡتِ ٱلۡقَوۡمَ ٱلظَّـٰلِمِینَ
11
قَوۡمَ فِرۡعَوۡنَۚ أَلَا یَتَّقُونَ
12
قَالَ رَبِّ إِنِّیۤ أَخَافُ أَن یُكَذِّبُونِ
13
وَیَضِیقُ صَدۡرِی وَلَا یَنطَلِقُ لِسَانِی فَأَرۡسِلۡ إِلَىٰ هَـٰرُونَ
14
وَلَهُمۡ عَلَیَّ ذَنۢبࣱ فَأَخَافُ أَن یَقۡتُلُونِ
15
قَالَ كَلَّاۖ فَٱذۡهَبَا بِـَٔایَـٰتِنَاۤۖ إِنَّا مَعَكُم مُّسۡتَمِعُونَ
16
فَأۡتِیَا فِرۡعَوۡنَ فَقُولَاۤ إِنَّا رَسُولُ رَبِّ ٱلۡعَـٰلَمِینَ
17
أَنۡ أَرۡسِلۡ مَعَنَا بَنِیۤ إِسۡرَ ٰ⁠ۤءِیلَ
18
قَالَ أَلَمۡ نُرَبِّكَ فِینَا وَلِیدࣰا وَلَبِثۡتَ فِینَا مِنۡ عُمُرِكَ سِنِینَ
19
وَفَعَلۡتَ فَعۡلَتَكَ ٱلَّتِی فَعَلۡتَ وَأَنتَ مِنَ ٱلۡكَـٰفِرِینَ
20
قَالَ فَعَلۡتُهَاۤ إِذࣰا وَأَنَا۠ مِنَ ٱلضَّاۤلِّینَ
21
فَفَرَرۡتُ مِنكُمۡ لَمَّا خِفۡتُكُمۡ فَوَهَبَ لِی رَبِّی حُكۡمࣰا وَجَعَلَنِی مِنَ ٱلۡمُرۡسَلِینَ
22
وَتِلۡكَ نِعۡمَةࣱ تَمُنُّهَا عَلَیَّ أَنۡ عَبَّدتَّ بَنِیۤ إِسۡرَ ٰ⁠ۤءِیلَ
23
قَالَ فِرۡعَوۡنُ وَمَا رَبُّ ٱلۡعَـٰلَمِینَ
24
قَالَ رَبُّ ٱلسَّمَـٰوَ ٰ⁠تِ وَٱلۡأَرۡضِ وَمَا بَیۡنَهُمَاۤۖ إِن كُنتُم مُّوقِنِینَ
25
قَالَ لِمَنۡ حَوۡلَهُۥۤ أَلَا تَسۡتَمِعُونَ
26
قَالَ رَبُّكُمۡ وَرَبُّ ءَابَاۤىِٕكُمُ ٱلۡأَوَّلِینَ
27
قَالَ إِنَّ رَسُولَكُمُ ٱلَّذِیۤ أُرۡسِلَ إِلَیۡكُمۡ لَمَجۡنُونࣱ
28
قَالَ رَبُّ ٱلۡمَشۡرِقِ وَٱلۡمَغۡرِبِ وَمَا بَیۡنَهُمَاۤۖ إِن كُنتُمۡ تَعۡقِلُونَ
29
قَالَ لَىِٕنِ ٱتَّخَذۡتَ إِلَـٰهًا غَیۡرِی لَأَجۡعَلَنَّكَ مِنَ ٱلۡمَسۡجُونِینَ
30
قَالَ أَوَلَوۡ جِئۡتُكَ بِشَیۡءࣲ مُّبِینࣲ
31
قَالَ فَأۡتِ بِهِۦۤ إِن كُنتَ مِنَ ٱلصَّـٰدِقِینَ
32
فَأَلۡقَىٰ عَصَاهُ فَإِذَا هِیَ ثُعۡبَانࣱ مُّبِینࣱ
33
وَنَزَعَ یَدَهُۥ فَإِذَا هِیَ بَیۡضَاۤءُ لِلنَّـٰظِرِینَ
34
قَالَ لِلۡمَلَإِ حَوۡلَهُۥۤ إِنَّ هَـٰذَا لَسَـٰحِرٌ عَلِیمࣱ
35
یُرِیدُ أَن یُخۡرِجَكُم مِّنۡ أَرۡضِكُم بِسِحۡرِهِۦ فَمَاذَا تَأۡمُرُونَ
36
قَالُوۤا۟ أَرۡجِهۡ وَأَخَاهُ وَٱبۡعَثۡ فِی ٱلۡمَدَاۤىِٕنِ حَـٰشِرِینَ
37
یَأۡتُوكَ بِكُلِّ سَحَّارٍ عَلِیمࣲ
38
فَجُمِعَ ٱلسَّحَرَةُ لِمِیقَـٰتِ یَوۡمࣲ مَّعۡلُومࣲ
39
وَقِیلَ لِلنَّاسِ هَلۡ أَنتُم مُّجۡتَمِعُونَ
40
لَعَلَّنَا نَتَّبِعُ ٱلسَّحَرَةَ إِن كَانُوا۟ هُمُ ٱلۡغَـٰلِبِینَ
41
فَلَمَّا جَاۤءَ ٱلسَّحَرَةُ قَالُوا۟ لِفِرۡعَوۡنَ أَىِٕنَّ لَنَا لَأَجۡرًا إِن كُنَّا نَحۡنُ ٱلۡغَـٰلِبِینَ
42
قَالَ نَعَمۡ وَإِنَّكُمۡ إِذࣰا لَّمِنَ ٱلۡمُقَرَّبِینَ
43
قَالَ لَهُم مُّوسَىٰۤ أَلۡقُوا۟ مَاۤ أَنتُم مُّلۡقُونَ
44
فَأَلۡقَوۡا۟ حِبَالَهُمۡ وَعِصِیَّهُمۡ وَقَالُوا۟ بِعِزَّةِ فِرۡعَوۡنَ إِنَّا لَنَحۡنُ ٱلۡغَـٰلِبُونَ
45
فَأَلۡقَىٰ مُوسَىٰ عَصَاهُ فَإِذَا هِیَ تَلۡقَفُ مَا یَأۡفِكُونَ
46
فَأُلۡقِیَ ٱلسَّحَرَةُ سَـٰجِدِینَ
47
قَالُوۤا۟ ءَامَنَّا بِرَبِّ ٱلۡعَـٰلَمِینَ
48
رَبِّ مُوسَىٰ وَهَـٰرُونَ
49
قَالَ ءَامَنتُمۡ لَهُۥ قَبۡلَ أَنۡ ءَاذَنَ لَكُمۡۖ إِنَّهُۥ لَكَبِیرُكُمُ ٱلَّذِی عَلَّمَكُمُ ٱلسِّحۡرَ فَلَسَوۡفَ تَعۡلَمُونَۚ لَأُقَطِّعَنَّ أَیۡدِیَكُمۡ وَأَرۡجُلَكُم مِّنۡ خِلَـٰفࣲ وَلَأُصَلِّبَنَّكُمۡ أَجۡمَعِینَ
50
قَالُوا۟ لَا ضَیۡرَۖ إِنَّاۤ إِلَىٰ رَبِّنَا مُنقَلِبُونَ
51
إِنَّا نَطۡمَعُ أَن یَغۡفِرَ لَنَا رَبُّنَا خَطَـٰیَـٰنَاۤ أَن كُنَّاۤ أَوَّلَ ٱلۡمُؤۡمِنِینَ
52
۞ وَأَوۡحَیۡنَاۤ إِلَىٰ مُوسَىٰۤ أَنۡ أَسۡرِ بِعِبَادِیۤ إِنَّكُم مُّتَّبَعُونَ
53
فَأَرۡسَلَ فِرۡعَوۡنُ فِی ٱلۡمَدَاۤىِٕنِ حَـٰشِرِینَ
54
إِنَّ هَـٰۤؤُلَاۤءِ لَشِرۡذِمَةࣱ قَلِیلُونَ
55
وَإِنَّهُمۡ لَنَا لَغَاۤىِٕظُونَ
56
وَإِنَّا لَجَمِیعٌ حَـٰذِرُونَ
57
فَأَخۡرَجۡنَـٰهُم مِّن جَنَّـٰتࣲ وَعُیُونࣲ
58
وَكُنُوزࣲ وَمَقَامࣲ كَرِیمࣲ
59
كَذَ ٰ⁠لِكَۖ وَأَوۡرَثۡنَـٰهَا بَنِیۤ إِسۡرَ ٰ⁠ۤءِیلَ
60
فَأَتۡبَعُوهُم مُّشۡرِقِینَ
61
فَلَمَّا تَرَ ٰ⁠ۤءَا ٱلۡجَمۡعَانِ قَالَ أَصۡحَـٰبُ مُوسَىٰۤ إِنَّا لَمُدۡرَكُونَ
62
قَالَ كَلَّاۤۖ إِنَّ مَعِیَ رَبِّی سَیَهۡدِینِ
63
فَأَوۡحَیۡنَاۤ إِلَىٰ مُوسَىٰۤ أَنِ ٱضۡرِب بِّعَصَاكَ ٱلۡبَحۡرَۖ فَٱنفَلَقَ فَكَانَ كُلُّ فِرۡقࣲ كَٱلطَّوۡدِ ٱلۡعَظِیمِ
64
وَأَزۡلَفۡنَا ثَمَّ ٱلۡـَٔاخَرِینَ
65
وَأَنجَیۡنَا مُوسَىٰ وَمَن مَّعَهُۥۤ أَجۡمَعِینَ
66
ثُمَّ أَغۡرَقۡنَا ٱلۡـَٔاخَرِینَ
67
إِنَّ فِی ذَ ٰ⁠لِكَ لَـَٔایَةࣰۖ وَمَا كَانَ أَكۡثَرُهُم مُّؤۡمِنِینَ
68
وَإِنَّ رَبَّكَ لَهُوَ ٱلۡعَزِیزُ ٱلرَّحِیمُ
69
وَٱتۡلُ عَلَیۡهِمۡ نَبَأَ إِبۡرَ ٰ⁠هِیمَ
70
إِذۡ قَالَ لِأَبِیهِ وَقَوۡمِهِۦ مَا تَعۡبُدُونَ
71
قَالُوا۟ نَعۡبُدُ أَصۡنَامࣰا فَنَظَلُّ لَهَا عَـٰكِفِینَ
72
قَالَ هَلۡ یَسۡمَعُونَكُمۡ إِذۡ تَدۡعُونَ
73
أَوۡ یَنفَعُونَكُمۡ أَوۡ یَضُرُّونَ
74
قَالُوا۟ بَلۡ وَجَدۡنَاۤ ءَابَاۤءَنَا كَذَ ٰ⁠لِكَ یَفۡعَلُونَ
75
قَالَ أَفَرَءَیۡتُم مَّا كُنتُمۡ تَعۡبُدُونَ
76
أَنتُمۡ وَءَابَاۤؤُكُمُ ٱلۡأَقۡدَمُونَ
77
فَإِنَّهُمۡ عَدُوࣱّ لِّیۤ إِلَّا رَبَّ ٱلۡعَـٰلَمِینَ
78
ٱلَّذِی خَلَقَنِی فَهُوَ یَهۡدِینِ
79
وَٱلَّذِی هُوَ یُطۡعِمُنِی وَیَسۡقِینِ
80
وَإِذَا مَرِضۡتُ فَهُوَ یَشۡفِینِ
81
وَٱلَّذِی یُمِیتُنِی ثُمَّ یُحۡیِینِ
82
وَٱلَّذِیۤ أَطۡمَعُ أَن یَغۡفِرَ لِی خَطِیۤـَٔتِی یَوۡمَ ٱلدِّینِ
83
رَبِّ هَبۡ لِی حُكۡمࣰا وَأَلۡحِقۡنِی بِٱلصَّـٰلِحِینَ
84
وَٱجۡعَل لِّی لِسَانَ صِدۡقࣲ فِی ٱلۡـَٔاخِرِینَ
85
وَٱجۡعَلۡنِی مِن وَرَثَةِ جَنَّةِ ٱلنَّعِیمِ
86
وَٱغۡفِرۡ لِأَبِیۤ إِنَّهُۥ كَانَ مِنَ ٱلضَّاۤلِّینَ
87
وَلَا تُخۡزِنِی یَوۡمَ یُبۡعَثُونَ
88
یَوۡمَ لَا یَنفَعُ مَالࣱ وَلَا بَنُونَ
89
إِلَّا مَنۡ أَتَى ٱللَّهَ بِقَلۡبࣲ سَلِیمࣲ
90
وَأُزۡلِفَتِ ٱلۡجَنَّةُ لِلۡمُتَّقِینَ
91
وَبُرِّزَتِ ٱلۡجَحِیمُ لِلۡغَاوِینَ
92
وَقِیلَ لَهُمۡ أَیۡنَ مَا كُنتُمۡ تَعۡبُدُونَ
93
مِن دُونِ ٱللَّهِ هَلۡ یَنصُرُونَكُمۡ أَوۡ یَنتَصِرُونَ
94
فَكُبۡكِبُوا۟ فِیهَا هُمۡ وَٱلۡغَاوُۥنَ
95
وَجُنُودُ إِبۡلِیسَ أَجۡمَعُونَ
96
قَالُوا۟ وَهُمۡ فِیهَا یَخۡتَصِمُونَ
97
تَٱللَّهِ إِن كُنَّا لَفِی ضَلَـٰلࣲ مُّبِینٍ
98
إِذۡ نُسَوِّیكُم بِرَبِّ ٱلۡعَـٰلَمِینَ
99
وَمَاۤ أَضَلَّنَاۤ إِلَّا ٱلۡمُجۡرِمُونَ
100
فَمَا لَنَا مِن شَـٰفِعِینَ
101
وَلَا صَدِیقٍ حَمِیمࣲ
102
فَلَوۡ أَنَّ لَنَا كَرَّةࣰ فَنَكُونَ مِنَ ٱلۡمُؤۡمِنِینَ
103
إِنَّ فِی ذَ ٰ⁠لِكَ لَـَٔایَةࣰۖ وَمَا كَانَ أَكۡثَرُهُم مُّؤۡمِنِینَ
104
وَإِنَّ رَبَّكَ لَهُوَ ٱلۡعَزِیزُ ٱلرَّحِیمُ
105
كَذَّبَتۡ قَوۡمُ نُوحٍ ٱلۡمُرۡسَلِینَ
106
إِذۡ قَالَ لَهُمۡ أَخُوهُمۡ نُوحٌ أَلَا تَتَّقُونَ
107
إِنِّی لَكُمۡ رَسُولٌ أَمِینࣱ
108
فَٱتَّقُوا۟ ٱللَّهَ وَأَطِیعُونِ
109
وَمَاۤ أَسۡـَٔلُكُمۡ عَلَیۡهِ مِنۡ أَجۡرٍۖ إِنۡ أَجۡرِیَ إِلَّا عَلَىٰ رَبِّ ٱلۡعَـٰلَمِینَ
110
فَٱتَّقُوا۟ ٱللَّهَ وَأَطِیعُونِ
111
۞ قَالُوۤا۟ أَنُؤۡمِنُ لَكَ وَٱتَّبَعَكَ ٱلۡأَرۡذَلُونَ
112
قَالَ وَمَا عِلۡمِی بِمَا كَانُوا۟ یَعۡمَلُونَ
113
إِنۡ حِسَابُهُمۡ إِلَّا عَلَىٰ رَبِّیۖ لَوۡ تَشۡعُرُونَ
114
وَمَاۤ أَنَا۠ بِطَارِدِ ٱلۡمُؤۡمِنِینَ
115
إِنۡ أَنَا۠ إِلَّا نَذِیرࣱ مُّبِینࣱ
116
قَالُوا۟ لَىِٕن لَّمۡ تَنتَهِ یَـٰنُوحُ لَتَكُونَنَّ مِنَ ٱلۡمَرۡجُومِینَ
117
قَالَ رَبِّ إِنَّ قَوۡمِی كَذَّبُونِ
118
فَٱفۡتَحۡ بَیۡنِی وَبَیۡنَهُمۡ فَتۡحࣰا وَنَجِّنِی وَمَن مَّعِیَ مِنَ ٱلۡمُؤۡمِنِینَ
119
فَأَنجَیۡنَـٰهُ وَمَن مَّعَهُۥ فِی ٱلۡفُلۡكِ ٱلۡمَشۡحُونِ
120
ثُمَّ أَغۡرَقۡنَا بَعۡدُ ٱلۡبَاقِینَ
121
إِنَّ فِی ذَ ٰ⁠لِكَ لَـَٔایَةࣰۖ وَمَا كَانَ أَكۡثَرُهُم مُّؤۡمِنِینَ
122
وَإِنَّ رَبَّكَ لَهُوَ ٱلۡعَزِیزُ ٱلرَّحِیمُ
123
كَذَّبَتۡ عَادٌ ٱلۡمُرۡسَلِینَ
124
إِذۡ قَالَ لَهُمۡ أَخُوهُمۡ هُودٌ أَلَا تَتَّقُونَ
125
إِنِّی لَكُمۡ رَسُولٌ أَمِینࣱ
126
فَٱتَّقُوا۟ ٱللَّهَ وَأَطِیعُونِ
127
وَمَاۤ أَسۡـَٔلُكُمۡ عَلَیۡهِ مِنۡ أَجۡرٍۖ إِنۡ أَجۡرِیَ إِلَّا عَلَىٰ رَبِّ ٱلۡعَـٰلَمِینَ
128
أَتَبۡنُونَ بِكُلِّ رِیعٍ ءَایَةࣰ تَعۡبَثُونَ
129
وَتَتَّخِذُونَ مَصَانِعَ لَعَلَّكُمۡ تَخۡلُدُونَ
130
وَإِذَا بَطَشۡتُم بَطَشۡتُمۡ جَبَّارِینَ
131
فَٱتَّقُوا۟ ٱللَّهَ وَأَطِیعُونِ
132
وَٱتَّقُوا۟ ٱلَّذِیۤ أَمَدَّكُم بِمَا تَعۡلَمُونَ
133
أَمَدَّكُم بِأَنۡعَـٰمࣲ وَبَنِینَ
134
وَجَنَّـٰتࣲ وَعُیُونٍ
135
إِنِّیۤ أَخَافُ عَلَیۡكُمۡ عَذَابَ یَوۡمٍ عَظِیمࣲ
136
قَالُوا۟ سَوَاۤءٌ عَلَیۡنَاۤ أَوَعَظۡتَ أَمۡ لَمۡ تَكُن مِّنَ ٱلۡوَ ٰ⁠عِظِینَ
137
إِنۡ هَـٰذَاۤ إِلَّا خُلُقُ ٱلۡأَوَّلِینَ
138
وَمَا نَحۡنُ بِمُعَذَّبِینَ
139
فَكَذَّبُوهُ فَأَهۡلَكۡنَـٰهُمۡۚ إِنَّ فِی ذَ ٰ⁠لِكَ لَـَٔایَةࣰۖ وَمَا كَانَ أَكۡثَرُهُم مُّؤۡمِنِینَ
140
وَإِنَّ رَبَّكَ لَهُوَ ٱلۡعَزِیزُ ٱلرَّحِیمُ
141
كَذَّبَتۡ ثَمُودُ ٱلۡمُرۡسَلِینَ
142
إِذۡ قَالَ لَهُمۡ أَخُوهُمۡ صَـٰلِحٌ أَلَا تَتَّقُونَ
143
إِنِّی لَكُمۡ رَسُولٌ أَمِینࣱ
144
فَٱتَّقُوا۟ ٱللَّهَ وَأَطِیعُونِ
145
وَمَاۤ أَسۡـَٔلُكُمۡ عَلَیۡهِ مِنۡ أَجۡرٍۖ إِنۡ أَجۡرِیَ إِلَّا عَلَىٰ رَبِّ ٱلۡعَـٰلَمِینَ
146
أَتُتۡرَكُونَ فِی مَا هَـٰهُنَاۤ ءَامِنِینَ
147
فِی جَنَّـٰتࣲ وَعُیُونࣲ
148
وَزُرُوعࣲ وَنَخۡلࣲ طَلۡعُهَا هَضِیمࣱ
149
وَتَنۡحِتُونَ مِنَ ٱلۡجِبَالِ بُیُوتࣰا فَـٰرِهِینَ
150
فَٱتَّقُوا۟ ٱللَّهَ وَأَطِیعُونِ
151
وَلَا تُطِیعُوۤا۟ أَمۡرَ ٱلۡمُسۡرِفِینَ
152
ٱلَّذِینَ یُفۡسِدُونَ فِی ٱلۡأَرۡضِ وَلَا یُصۡلِحُونَ
153
قَالُوۤا۟ إِنَّمَاۤ أَنتَ مِنَ ٱلۡمُسَحَّرِینَ
154
مَاۤ أَنتَ إِلَّا بَشَرࣱ مِّثۡلُنَا فَأۡتِ بِـَٔایَةٍ إِن كُنتَ مِنَ ٱلصَّـٰدِقِینَ
155
قَالَ هَـٰذِهِۦ نَاقَةࣱ لَّهَا شِرۡبࣱ وَلَكُمۡ شِرۡبُ یَوۡمࣲ مَّعۡلُومࣲ
156
وَلَا تَمَسُّوهَا بِسُوۤءࣲ فَیَأۡخُذَكُمۡ عَذَابُ یَوۡمٍ عَظِیمࣲ
157
فَعَقَرُوهَا فَأَصۡبَحُوا۟ نَـٰدِمِینَ
158
فَأَخَذَهُمُ ٱلۡعَذَابُۚ إِنَّ فِی ذَ ٰ⁠لِكَ لَـَٔایَةࣰۖ وَمَا كَانَ أَكۡثَرُهُم مُّؤۡمِنِینَ
159
وَإِنَّ رَبَّكَ لَهُوَ ٱلۡعَزِیزُ ٱلرَّحِیمُ
160
كَذَّبَتۡ قَوۡمُ لُوطٍ ٱلۡمُرۡسَلِینَ
161
إِذۡ قَالَ لَهُمۡ أَخُوهُمۡ لُوطٌ أَلَا تَتَّقُونَ
162
إِنِّی لَكُمۡ رَسُولٌ أَمِینࣱ
163
فَٱتَّقُوا۟ ٱللَّهَ وَأَطِیعُونِ
164
وَمَاۤ أَسۡـَٔلُكُمۡ عَلَیۡهِ مِنۡ أَجۡرٍۖ إِنۡ أَجۡرِیَ إِلَّا عَلَىٰ رَبِّ ٱلۡعَـٰلَمِینَ
165
أَتَأۡتُونَ ٱلذُّكۡرَانَ مِنَ ٱلۡعَـٰلَمِینَ
166
وَتَذَرُونَ مَا خَلَقَ لَكُمۡ رَبُّكُم مِّنۡ أَزۡوَ ٰ⁠جِكُمۚ بَلۡ أَنتُمۡ قَوۡمٌ عَادُونَ
167
قَالُوا۟ لَىِٕن لَّمۡ تَنتَهِ یَـٰلُوطُ لَتَكُونَنَّ مِنَ ٱلۡمُخۡرَجِینَ
168
قَالَ إِنِّی لِعَمَلِكُم مِّنَ ٱلۡقَالِینَ
169
رَبِّ نَجِّنِی وَأَهۡلِی مِمَّا یَعۡمَلُونَ
170
فَنَجَّیۡنَـٰهُ وَأَهۡلَهُۥۤ أَجۡمَعِینَ
171
إِلَّا عَجُوزࣰا فِی ٱلۡغَـٰبِرِینَ
172
ثُمَّ دَمَّرۡنَا ٱلۡـَٔاخَرِینَ
173
وَأَمۡطَرۡنَا عَلَیۡهِم مَّطَرࣰاۖ فَسَاۤءَ مَطَرُ ٱلۡمُنذَرِینَ
174
إِنَّ فِی ذَ ٰ⁠لِكَ لَـَٔایَةࣰۖ وَمَا كَانَ أَكۡثَرُهُم مُّؤۡمِنِینَ
175
وَإِنَّ رَبَّكَ لَهُوَ ٱلۡعَزِیزُ ٱلرَّحِیمُ
176
كَذَّبَ أَصۡحَـٰبُ لۡـَٔیۡكَةِ ٱلۡمُرۡسَلِینَ
177
إِذۡ قَالَ لَهُمۡ شُعَیۡبٌ أَلَا تَتَّقُونَ
178
إِنِّی لَكُمۡ رَسُولٌ أَمِینࣱ
179
فَٱتَّقُوا۟ ٱللَّهَ وَأَطِیعُونِ
180
وَمَاۤ أَسۡـَٔلُكُمۡ عَلَیۡهِ مِنۡ أَجۡرٍۖ إِنۡ أَجۡرِیَ إِلَّا عَلَىٰ رَبِّ ٱلۡعَـٰلَمِینَ
181
۞ أَوۡفُوا۟ ٱلۡكَیۡلَ وَلَا تَكُونُوا۟ مِنَ ٱلۡمُخۡسِرِینَ
182
وَزِنُوا۟ بِٱلۡقِسۡطَاسِ ٱلۡمُسۡتَقِیمِ
183
وَلَا تَبۡخَسُوا۟ ٱلنَّاسَ أَشۡیَاۤءَهُمۡ وَلَا تَعۡثَوۡا۟ فِی ٱلۡأَرۡضِ مُفۡسِدِینَ
184
وَٱتَّقُوا۟ ٱلَّذِی خَلَقَكُمۡ وَٱلۡجِبِلَّةَ ٱلۡأَوَّلِینَ
185
قَالُوۤا۟ إِنَّمَاۤ أَنتَ مِنَ ٱلۡمُسَحَّرِینَ
186
وَمَاۤ أَنتَ إِلَّا بَشَرࣱ مِّثۡلُنَا وَإِن نَّظُنُّكَ لَمِنَ ٱلۡكَـٰذِبِینَ
187
فَأَسۡقِطۡ عَلَیۡنَا كِسَفࣰا مِّنَ ٱلسَّمَاۤءِ إِن كُنتَ مِنَ ٱلصَّـٰدِقِینَ
188
قَالَ رَبِّیۤ أَعۡلَمُ بِمَا تَعۡمَلُونَ
189
فَكَذَّبُوهُ فَأَخَذَهُمۡ عَذَابُ یَوۡمِ ٱلظُّلَّةِۚ إِنَّهُۥ كَانَ عَذَابَ یَوۡمٍ عَظِیمٍ
190
إِنَّ فِی ذَ ٰ⁠لِكَ لَـَٔایَةࣰۖ وَمَا كَانَ أَكۡثَرُهُم مُّؤۡمِنِینَ
191
وَإِنَّ رَبَّكَ لَهُوَ ٱلۡعَزِیزُ ٱلرَّحِیمُ
192
وَإِنَّهُۥ لَتَنزِیلُ رَبِّ ٱلۡعَـٰلَمِینَ
193
نَزَلَ بِهِ ٱلرُّوحُ ٱلۡأَمِینُ
194
عَلَىٰ قَلۡبِكَ لِتَكُونَ مِنَ ٱلۡمُنذِرِینَ
195
بِلِسَانٍ عَرَبِیࣲّ مُّبِینࣲ
196
وَإِنَّهُۥ لَفِی زُبُرِ ٱلۡأَوَّلِینَ
197
أَوَلَمۡ یَكُن لَّهُمۡ ءَایَةً أَن یَعۡلَمَهُۥ عُلَمَـٰۤؤُا۟ بَنِیۤ إِسۡرَ ٰ⁠ۤءِیلَ
198
وَلَوۡ نَزَّلۡنَـٰهُ عَلَىٰ بَعۡضِ ٱلۡأَعۡجَمِینَ
199
فَقَرَأَهُۥ عَلَیۡهِم مَّا كَانُوا۟ بِهِۦ مُؤۡمِنِینَ
200
كَذَ ٰ⁠لِكَ سَلَكۡنَـٰهُ فِی قُلُوبِ ٱلۡمُجۡرِمِینَ
201
لَا یُؤۡمِنُونَ بِهِۦ حَتَّىٰ یَرَوُا۟ ٱلۡعَذَابَ ٱلۡأَلِیمَ
202
فَیَأۡتِیَهُم بَغۡتَةࣰ وَهُمۡ لَا یَشۡعُرُونَ
203
فَیَقُولُوا۟ هَلۡ نَحۡنُ مُنظَرُونَ
204
أَفَبِعَذَابِنَا یَسۡتَعۡجِلُونَ
205
أَفَرَءَیۡتَ إِن مَّتَّعۡنَـٰهُمۡ سِنِینَ
206
ثُمَّ جَاۤءَهُم مَّا كَانُوا۟ یُوعَدُونَ
207
مَاۤ أَغۡنَىٰ عَنۡهُم مَّا كَانُوا۟ یُمَتَّعُونَ
208
وَمَاۤ أَهۡلَكۡنَا مِن قَرۡیَةٍ إِلَّا لَهَا مُنذِرُونَ
209
ذِكۡرَىٰ وَمَا كُنَّا ظَـٰلِمِینَ
210
وَمَا تَنَزَّلَتۡ بِهِ ٱلشَّیَـٰطِینُ
211
وَمَا یَنۢبَغِی لَهُمۡ وَمَا یَسۡتَطِیعُونَ
212
إِنَّهُمۡ عَنِ ٱلسَّمۡعِ لَمَعۡزُولُونَ
213
فَلَا تَدۡعُ مَعَ ٱللَّهِ إِلَـٰهًا ءَاخَرَ فَتَكُونَ مِنَ ٱلۡمُعَذَّبِینَ
214
وَأَنذِرۡ عَشِیرَتَكَ ٱلۡأَقۡرَبِینَ
215
وَٱخۡفِضۡ جَنَاحَكَ لِمَنِ ٱتَّبَعَكَ مِنَ ٱلۡمُؤۡمِنِینَ
216
فَإِنۡ عَصَوۡكَ فَقُلۡ إِنِّی بَرِیۤءࣱ مِّمَّا تَعۡمَلُونَ
217
وَتَوَكَّلۡ عَلَى ٱلۡعَزِیزِ ٱلرَّحِیمِ
218
ٱلَّذِی یَرَىٰكَ حِینَ تَقُومُ
219
وَتَقَلُّبَكَ فِی ٱلسَّـٰجِدِینَ
220
إِنَّهُۥ هُوَ ٱلسَّمِیعُ ٱلۡعَلِیمُ
221
هَلۡ أُنَبِّئُكُمۡ عَلَىٰ مَن تَنَزَّلُ ٱلشَّیَـٰطِینُ
222
تَنَزَّلُ عَلَىٰ كُلِّ أَفَّاكٍ أَثِیمࣲ
223
یُلۡقُونَ ٱلسَّمۡعَ وَأَكۡثَرُهُمۡ كَـٰذِبُونَ
224
وَٱلشُّعَرَاۤءُ یَتَّبِعُهُمُ ٱلۡغَاوُۥنَ
225
أَلَمۡ تَرَ أَنَّهُمۡ فِی كُلِّ وَادࣲ یَهِیمُونَ
226
وَأَنَّهُمۡ یَقُولُونَ مَا لَا یَفۡعَلُونَ
227
إِلَّا ٱلَّذِینَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّـٰلِحَـٰتِ وَذَكَرُوا۟ ٱللَّهَ كَثِیرࣰا وَٱنتَصَرُوا۟ مِنۢ بَعۡدِ مَا ظُلِمُوا۟ۗ وَسَیَعۡلَمُ ٱلَّذِینَ ظَلَمُوۤا۟ أَیَّ مُنقَلَبࣲ یَنقَلِبُونَ
